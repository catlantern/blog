## 1. 上下文切换概述

### 1.1 什么是上下文切换？

```
上下文切换定义：
+===========================================================================+
|                                                                           |
|   上下文切换 = 保存当前进程的状态，恢复另一个进程的状态                   |
|                                                                           |
|   进程上下文包括：                                                        |
|   +------------------------------------------------------------------+    |
|   |                                                                   |    |
|   |   1. CPU 寄存器                                                   |    |
|   |      - 通用寄存器：eax, ebx, ecx, edx, esi, edi, ebp, esp        |    |
|   |      - 段寄存器：cs, ds, es, fs, gs, ss                          |    |
|   |      - 标志寄存器：eflags                                         |    |
|   |      - 指令指针：eip                                              |    |
|   |                                                                   |    |
|   |   2. 内存管理信息                                                 |    |
|   |      - 页表基址（CR3）                                            |    |
|   |      - LDT（局部描述符表）                                        |    |
|   |                                                                   |    |
|   |   3. 内核栈                                                       |    |
|   |      - 每个进程有独立的内核栈                                     |    |
|   |                                                                   |    |
|   |   4. 浮点寄存器（FPU）                                            |    |
|   |      - 如果进程使用了浮点运算                                     |    |
|   |                                                                   |    |
|   +------------------------------------------------------------------+    |
|                                                                           |
+===========================================================================+
```

### 1.2 上下文切换的时机

```
上下文切换的时机：
+===========================================================================+
|                                                                           |
|   1. 时间片用完                                                           |
|      +------------------+                                                 |
|      | 时钟中断         |  current->counter <= 0                         |
|      +--------+---------+                                                 |
|               |                                                           |
|               v                                                           |
|      +------------------+                                                 |
|      | schedule()       |  选择新进程运行                                 |
|      +------------------+                                                 |
|                                                                           |
|   2. 进程主动放弃 CPU                                                     |
|      +------------------+                                                 |
|      | sleep_on()       |  等待资源                                       |
|      | pause()          |  主动暂停                                       |
|      | exit()           |  进程退出                                       |
|      +------------------+                                                 |
|                                                                           |
|   3. 系统调用返回时                                                       |
|      +------------------+                                                 |
|      | ret_from_sys_call|  检查是否需要调度                               |
|      +------------------+                                                 |
|                                                                           |
+===========================================================================+
```

### 2.2 TSS 的作用

```
TSS（Task State Segment）的作用：
+===========================================================================+
|                                                                           |
```

```c
    int tty;
    unsigned short umask;
    struct m_inode * pwd, * root, * executable;
```

## 4. 调度器 schedule()

### 4.1 schedule() 函数

定义在 **`kernel/sched.c`**：

```c
void schedule(void)
{
    int i, next, c;
    struct task_struct ** p;

    /* 检查告警信号，唤醒可中断睡眠的进程 */
    for(p = &LAST_TASK; p > &FIRST_TASK; --p)
        if (*p) {
            /* 检查告警定时器 */
            if ((*p)->alarm && (*p)->alarm < jiffies) {
                (*p)->signal |= (1<<(SIGALRM-1));
                (*p)->alarm = 0;
            }
            /* 如果进程收到信号且处于可中断睡眠，则唤醒 */
            if (((*p)->signal & ~(_BLOCKABLE & (*p)->blocked)) &&
                (*p)->state == TASK_INTERRUPTIBLE)
                (*p)->state = TASK_RUNNING;
        }

    /* 调度算法：选择时间片最大的就绪进程 */
    while (1) {
        c = -1;
        next = 0;
        i = NR_TASKS;
        p = &task[NR_TASKS];
        
        /* 遍历所有进程，找出时间片最大的就绪进程 */
        while (--i) {
            if (!*--p)
                continue;
            if ((*p)->state == TASK_RUNNING && (*p)->counter > c)
                c = (*p)->counter, next = i;
        }
        
        /* 如果找到可运行的进程，跳出循环 */
        if (c) break;
        
        /* 所有就绪进程的时间片都用完了，重新分配时间片 */
        for(p = &LAST_TASK; p > &FIRST_TASK; --p)
            if (*p)
                (*p)->counter = ((*p)->counter >> 1) + (*p)->priority;
    }
    
    /* 切换到选中的进程 */
    switch_to(next);
}
```

## 5. switch_to 宏详解

### 5.1 switch_to 宏定义

定义在 **`include/linux/sched.h`**：

```c
#define switch_to(n) {\
struct {long a,b;} __tmp; \
__asm__("cmpl %%ecx,_current\n\t" \
    "je 1f\n\t" \
    "movw %%dx,%1\n\t" \
    "xchgl %%ecx,_current\n\t" \
    "ljmp %0\n\t" \
    "cmpl %%ecx,_last_task_used_math\n\t" \
    "jne 1f\n\t" \
    "clts\n" \
    "1:" \
    ::"m" (*&__tmp.a),"m" (*&__tmp.b), \
    "d" (_TSS(n)),"c" ((long) task[n])); \
}
```

### 5.2 switch_to 宏详解

```
switch_to 宏执行流程：
+===========================================================================+
|                                                                           |
|   switch_to(n)  // n 是目标进程的任务号                                   |
|       |                                                                   |
|       v                                                                   |
|   +------------------+                                                    |
|   | cmpl %%ecx,_current |  检查目标进程是否就是当前进程                 |
|   +--------+---------+                                                    |
|            |                                                              |
|      +-----+-----+                                                        |
|      | 相等       | 不相等                                                |
|      v           v                                                        |
|   +--------+  +------------------+                                        |
|   | 跳到 1f|  | 继续执行切换     |                                        |
|   | 不切换 |  |                  |                                        |
|   +--------+  +--------+---------+                                        |
|                        |                                                  |
|                        v                                                  |
|              +------------------+                                         |
|              | movw %%dx,%1    |  设置 TSS 选择子到 __tmp.b              |
|              +--------+---------+                                         |
|                        |                                                  |
|                        v                                                  |
|              +------------------+                                         |
|              | xchgl %%ecx,_current |  更新 current 指针                 |
|              +--------+---------+                                         |
|                        |                                                  |
|                        v                                                  |
|              +------------------+                                         |
|              | ljmp %0         |  关键！跳转到目标 TSS                   |
|              +--------+---------+                                         |
|                        |                                                  |
|                        v                                                  |
|              +------------------+                                         |
|              | 硬件自动保存   |  保存当前进程的所有寄存器到其 TSS        |
|              | 当前进程状态   |                                          |
|              +--------+---------+                                         |
|                        |                                                  |
|                        v                                                  |
|              +------------------+                                         |
|              | 硬件自动恢复   |  从目标进程的 TSS 恢复所有寄存器        |
|              | 目标进程状态   |                                          |
|              +--------+---------+                                         |
|                        |                                                  |
|                        v                                                  |
|              +------------------+                                         |
|              | 检查 FPU 使用   |  如果目标进程使用过 FPU，清除 TS 标志  |
|              +--------+---------+                                         |
|                        |                                                  |
|                        v                                                  |
|                   1: (结束)                                               |
|                                                                           |
+===========================================================================+
```

### 5.3 ljmp 指令的作用

```
ljmp 指令（长跳转）：
+===========================================================================+
|                                                                           |
|   ljmp 指令格式：ljmp segment:offset                                       |
|                                                                           |
|   当 segment 是 TSS 描述符时：                                            |
|   +------------------------------------------------------------------+    |
|   |                                                                   |    |
|   |   1. 硬件自动保存当前 CPU 状态到当前 TSS                          |    |
|   |      - 保存所有通用寄存器                                        |    |
|   |      - 保存段寄存器                                              |    |
|   |      - 保存 eip, eflags, esp, ebp 等                             |    |
|   |      - 保存 CR3（页表基址）                                      |    |
|   |                                                                   |    |
|   |   2. 硬件自动加载目标 TSS 的内容到 CPU                            |    |
|   |      - 恢复所有通用寄存器                                        |    |
|   |      - 恢复段寄存器                                              |    |
|   |      - 恢复 eip, eflags, esp, ebp 等                             |    |
|   |      - 恢复 CR3（切换页表）                                      |    |
|   |                                                                   |    |
|   |   3. 跳转到目标进程的 eip 处继续执行                              |    |
|   |                                                                   |    |
|   +------------------------------------------------------------------+    |
|                                                                           |
|   这就是 x86 硬件提供的任务切换机制！                                     |
|                                                                           |
+===========================================================================+
```

### 5.4 TSS 选择子的计算

```c
/* TSS 在 GDT 中的位置 */
#define FIRST_TSS_ENTRY 4

/* 计算 TSS 选择子 */
#define _TSS(n) ((((unsigned long) n)<<4)+(FIRST_TSS_ENTRY<<3))

/*
 * 解释：
 * GDT 结构：
 *   0 - 空描述符
 *   1 - 内核代码段
 *   2 - 内核数据段
 *   3 - 系统调用门
 *   4 - TSS0 (进程 0 的 TSS)
 *   5 - LDT0 (进程 0 的 LDT)
 *   6 - TSS1 (进程 1 的 TSS)
 *   7 - LDT1 (进程 1 的 LDT)
 *   ...
 *
 * _TSS(n) = (n << 4) + (4 << 3)
 *         = n * 16 + 32
 *         = 第 n 个进程的 TSS 选择子
 */
```

## 6. 时钟中断与调度触发

### 6.1 时钟中断处理

定义在 **`kernel/system_call.s`**：

```asm
_timer_interrupt:
    push %ds        /* 保存段寄存器 */
    push %es
    push %fs
    pushl %edx      /* 保存通用寄存器 */
    pushl %ecx
    pushl %ebx
    pushl %eax
    
    movl $0x10,%eax /* 设置内核数据段 */
    mov %ax,%ds
    mov %ax,%es
    movl $0x17,%eax /* fs 指向用户数据段 */
    mov %ax,%fs
    
    incl _jiffies   /* 增加时钟计数 */
    
    movb $0x20,%al  /* 发送 EOI 给 8259A */
    outb %al,$0x20
    
    movl CS(%esp),%eax  /* 获取当前特权级 */
    andl $3,%eax
    pushl %eax
    call _do_timer      /* 调用 do_timer() */
    addl $4,%esp
    
    jmp ret_from_sys_call
```

### 6.2 do_timer() 函数

定义在 **`kernel/sched.c`**：

```c
void do_timer(long cpl)
{
    extern int beepcount;
    extern void sysbeepstop(void);

    /* 处理蜂鸣 */
    if (beepcount)
        if (!--beepcount)
            sysbeepstop();

    /* 更新进程时间统计 */
    if (cpl)
        current->utime++;   /* 用户态时间 */
    else
        current->stime++;   /* 内核态时间 */

    /* 处理定时器链表 */
    if (next_timer) {
        next_timer->jiffies--;
        while (next_timer && next_timer->jiffies <= 0) {
            void (*fn)(void);
            fn = next_timer->fn;
            next_timer->fn = NULL;
            next_timer = next_timer->next;
            (fn)();
        }
    }

    /* 处理软驱马达定时器 */
    if (current_DOR & 0xf0)
        do_floppy_timer();

    /* 检查时间片 */
    if ((--current->counter) > 0) 
        return;  /* 时间片未用完，返回 */
    
    current->counter = 0;
    
    /* 如果在内核态，不调度 */
    if (!cpl) 
        return;
    
    /* 时间片用完，调用调度器 */
    schedule();
}
```

### 8.3 sleep_on 和 wake_up

```c
/* 不可中断睡眠 */
void sleep_on(struct task_struct **p)
{
    struct task_struct *tmp;

    if (!p)
        return;
    if (current == &(init_task.task))
        panic("task[0] trying to sleep");
    
    tmp = *p;           /* 保存之前的等待进程 */
    *p = current;       /* 将当前进程加入等待队列 */
    current->state = TASK_UNINTERRUPTIBLE;
    
    schedule();         /* 调度其他进程运行 */
    
    /* 被唤醒后继续执行 */
    if (tmp)
        tmp->state = 0; /* 唤醒等待队列中的前一个进程 */
}

/* 可中断睡眠 */
void interruptible_sleep_on(struct task_struct **p)
{
    struct task_struct *tmp;

    if (!p)
        return;
    if (current == &(init_task.task))
        panic("task[0] trying to sleep");
    
    tmp = *p;
    *p = current;
repeat:
    current->state = TASK_INTERRUPTIBLE;
    schedule();
    
    /* 检查是否是正确的唤醒顺序 */
    if (*p && *p != current) {
        (**p).state = 0;
        goto repeat;
    }
    *p = NULL;
    if (tmp)
        tmp->state = 0;
}

/* 唤醒进程 */
void wake_up(struct task_struct **p)
{
    if (p && *p) {
        (**p).state = 0;  /* 设置为 TASK_RUNNING */
        *p = NULL;
    }
}
```

## 9. 总结

### 9.1 上下文切换关键点

| 组件 | 作用 |
|------|------|
| TSS | 保存/恢复 CPU 寄存器状态 |

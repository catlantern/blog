## 一、brk概述

### 1.1 什么是brk？

brk（break）是Linux内核提供的最古老的**堆内存管理系统调用**。它通过移动"程序中断点"（program break）来扩展或收缩进程的堆内存区域。

```
┌─────────────────────────────────────────────────────────────┐
│                    brk核心思想                              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  堆内存区域：                                                │
│  ┌─────────────────────────────────────────────────────┐   │
│  │                                                     │   │
│  │  低地址                                             │   │
│  │  ┌─────────────────────────────────────────────┐   │   │
│  │  │              已分配的堆内存                   │   │   │
│  │  └─────────────────────────────────────────────┘   │   │
│  │                    ↑                               │   │
│  │              程序中断点(brk)                        │   │
│  │                    │                               │   │
│  │  ┌─────────────────────────────────────────────┐   │   │
│  │  │              未分配的堆空间                   │   │   │
│  │  └─────────────────────────────────────────────┘   │   │
│  │                    ↑                               │   │
│  │              堆的上边界                            │   │
│  │                                                     │   │
│  │  brk()系统调用：移动程序中断点位置                   │   │
│  │  • 向上移动：扩展堆内存                             │   │
│  │  • 向下移动：收缩堆内存                             │   │
│  │                                                     │   │
│  └─────────────────────────────────────────────────────┘   │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 brk函数原型

```c
#include <unistd.h>

int brk(void *addr);
void *sbrk(intptr_t increment);
```

**参数详解**：

| 函数 | 参数 | 说明 |
|------|------|------|
| `brk()` | `addr` | 设置新的程序中断点地址 |
| `sbrk()` | `increment` | 增量值（正数扩展，负数收缩） |

**返回值**：

| 函数 | 成功 | 失败 |
|------|------|------|
| `brk()` | 返回0 | 返回-1，设置errno |
| `sbrk()` | 返回旧的程序中断点 | 返回(void *)-1 |

### 1.3 brk与sbrk的关系

```
┌─────────────────────────────────────────────────────────────┐
│                  brk与sbrk的关系                            │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  sbrk()是brk()的包装函数：                                   │
│                                                             │
│  void *sbrk(intptr_t increment)                             │
│  {                                                          │
│      void *old_brk = current_brk;                           │
│      void *new_brk = old_brk + increment;                   │
│                                                             │
│      if (brk(new_brk) == 0)                                 │
│          return old_brk;    // 返回旧地址                   │
│      else                                                   │
│          return (void *)-1; // 失败                         │
│  }                                                          │
│                                                             │
│  使用示例：                                                  │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ // 分配4096字节                                     │   │
│  │ void *mem = sbrk(4096);                             │   │
│  │                                                     │   │
│  │ // 释放4096字节                                     │   │
│  │ sbrk(-4096);                                        │   │
│  │                                                     │   │
│  │ // 获取当前brk位置                                  │   │
│  │ void *current = sbrk(0);                            │   │
│  └─────────────────────────────────────────────────────┘   │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 四、brk系统调用实现

### 4.1 brk系统调用入口

```c
/*
 * mm/mmap.c
 * brk系统调用定义
 */
SYSCALL_DEFINE1(brk, unsigned long, brk)
{
    unsigned long newbrk, oldbrk, origbrk;
    struct mm_struct *mm = current->mm;
    struct vm_area_struct *next;
    unsigned long min_brk;
    bool populate;
    bool downgraded = false;
    LIST_HEAD(uf);

    /* 获取内存管理信号量 */
    if (mmap_write_lock_killable(mm))
        return -EINTR;

    origbrk = mm->brk;

#ifdef CONFIG_COMPAT_BRK
    /*
     * CONFIG_COMPAT_BRK兼容性处理：
     * 如果brk被随机化，min_brk设为start_brk
     * 否则设为end_data
     */
    if (current->brk_randomized)
        min_brk = mm->start_brk;
    else
        min_brk = mm->end_data;
#else
    min_brk = mm->start_brk;
#endif
    
    /* 新brk不能小于最小值 */
    if (brk < min_brk)
        goto out;

    /* 检查资源限制（RLIMIT_DATA） */
    if (check_data_rlimit(rlimit(RLIMIT_DATA), brk, mm->start_brk,
                          mm->end_data, mm->start_data))
        goto out;

    /* 页对齐处理 */
    newbrk = PAGE_ALIGN(brk);
    oldbrk = PAGE_ALIGN(mm->brk);
    
    /* 如果页对齐后相同，只需更新brk值 */
    if (oldbrk == newbrk) {
        mm->brk = brk;
        goto success;
    }

    /* 处理堆收缩 */
    if (brk <= mm->brk) {
        int ret;
        mm->brk = brk;
        ret = __do_munmap(mm, newbrk, oldbrk-newbrk, &uf, true);
        if (ret < 0) {
            mm->brk = origbrk;
            goto out;
        } else if (ret == 1) {
            downgraded = true;
        }
        goto success;
    }

    /* 检查是否与现有mmap区域冲突 */
    next = find_vma(mm, oldbrk);
    if (next && newbrk + PAGE_SIZE > vm_start_gap(next))
        goto out;

    /* 执行堆扩展 */
    if (do_brk_flags(oldbrk, newbrk-oldbrk, 0, &uf) < 0)
        goto out;
    
    mm->brk = brk;

success:
    populate = newbrk > oldbrk && (mm->def_flags & VM_LOCKED) != 0;
    if (downgraded)
        mmap_read_unlock(mm);
    else
        mmap_write_unlock(mm);
    userfaultfd_unmap_complete(mm, &uf);
    if (populate)
        mm_populate(oldbrk, newbrk - oldbrk);
    return brk;

out:
    mmap_write_unlock(mm);
    return origbrk;
}
```

### 4.2 brk系统调用流程图

```
┌─────────────────────────────────────────────────────────────┐
│                    brk系统调用流程                           │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  用户调用 brk(new_brk)                                       │
│       │                                                     │
│       ↓                                                     │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ 获取mmap_lock写锁                                    │   │
│  └─────────────────────────────────────────────────────┘   │
│       │                                                     │
│       ↓                                                     │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ 计算min_brk（最小允许的brk值）                        │   │
│  │ • CONFIG_COMPAT_BRK: 根据随机化选择                  │   │
│  │ • 否则: min_brk = start_brk                         │   │
│  └─────────────────────────────────────────────────────┘   │
│       │                                                     │
│       ↓                                                     │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ 检查: new_brk >= min_brk ?                          │   │
│  └─────────────────────────────────────────────────────┘   │
│       │                                                     │
│       ├── 否 ──> 返回原brk值                              │
│       │                                                     │
│       ↓ 是                                                  │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ 检查RLIMIT_DATA资源限制                              │   │
│  └─────────────────────────────────────────────────────┘   │
│       │                                                     │
│       ├── 超限 ──> 返回原brk值                            │
│       │                                                     │
│       ↓ 未超限                                              │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ 页对齐处理：                                          │   │
│  │ newbrk = PAGE_ALIGN(new_brk)                         │   │
│  │ oldbrk = PAGE_ALIGN(mm->brk)                         │   │
│  └─────────────────────────────────────────────────────┘   │
│       │                                                     │
│       ↓                                                     │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ oldbrk == newbrk ?                                   │   │
│  └─────────────────────────────────────────────────────┘   │
│       │                                                     │
│       ├── 是 ──> 更新mm->brk，返回成功                    │
│       │                                                     │
│       ↓ 否                                                  │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ new_brk <= mm->brk ?  （堆收缩）                     │   │
```

## 五、do_brk_flags核心函数

### 5.1 do_brk_flags函数实现

```c
/*
 * mm/mmap.c
 * do_brk_flags - 实际执行堆内存分配的函数
 * 
 * 参数：
 *   addr   - 起始地址
 *   len    - 分配长度
 *   flags  - VM标志位
 *   uf     - userfaultfd列表
 * 
 * 返回：
 *   成功返回0，失败返回负错误码
 */
static int do_brk_flags(unsigned long addr, unsigned long len, 
                        unsigned long flags, struct list_head *uf)
{
    struct mm_struct *mm = current->mm;
    struct vm_area_struct *vma, *prev;
    struct rb_node **rb_link, *rb_parent;
    pgoff_t pgoff = addr >> PAGE_SHIFT;
    int error;
    unsigned long mapped_addr;

    /* 只接受VM_EXEC标志 */
    if ((flags & (~VM_EXEC)) != 0)
        return -EINVAL;
    
    /* 设置默认标志位 */
    flags |= VM_DATA_DEFAULT_FLAGS | VM_ACCOUNT | mm->def_flags;

    /* 获取未映射区域 */
    mapped_addr = get_unmapped_area(NULL, addr, len, 0, MAP_FIXED);
    if (IS_ERR_VALUE(mapped_addr))
        return mapped_addr;

    /* 检查mlock限制 */
    error = mlock_future_check(mm, mm->def_flags, len);
    if (error)
        return error;

    /* 清除旧的映射，设置VMA链接信息 */
    if (munmap_vma_range(mm, addr, len, &prev, &rb_link, &rb_parent, uf))
        return -ENOMEM;

    /* 检查地址空间限制 */
    if (!may_expand_vm(mm, flags, len >> PAGE_SHIFT))
        return -ENOMEM;

    /* 检查VMA数量限制 */
    if (mm->map_count > sysctl_max_map_count)
        return -ENOMEM;

    /* 安全检查 */
    if (security_vm_enough_memory_mm(mm, len >> PAGE_SHIFT))
        return -ENOMEM;

    /* 尝试合并相邻的VMA */
    vma = vma_merge(mm, prev, addr, addr + len, flags,
                    NULL, NULL, pgoff, NULL, NULL_VM_UFFD_CTX);
    if (vma)
        goto out;

    /* 无法合并，创建新的VMA */
    vma = vm_area_alloc(mm);
    if (!vma) {
        vm_unacct_memory(len >> PAGE_SHIFT);
        return -ENOMEM;
    }

    /* 初始化VMA */
    vma_set_anonymous(vma);
    vma->vm_start = addr;
    vma->vm_end = addr + len;
    vma->vm_pgoff = pgoff;
    vma->vm_flags = flags;
    vma->vm_page_prot = vm_get_page_prot(flags);
    
    /* 将VMA插入进程地址空间 */
    vma_link(mm, vma, prev, rb_link, rb_parent);

out:
    /* 性能事件通知 */
    perf_event_mmap(vma);
    
    /* 更新统计信息 */
    mm->total_vm += len >> PAGE_SHIFT;
    mm->data_vm += len >> PAGE_SHIFT;
    if (flags & VM_LOCKED)
        mm->locked_vm += (len >> PAGE_SHIFT);
    vma->vm_flags |= VM_SOFTDIRTY;
    
    return 0;
}
```

### 5.3 vm_brk_flags函数

```c
/*
 * mm/mmap.c
 * vm_brk_flags - 内核内部使用的brk接口
 */
int vm_brk_flags(unsigned long addr, unsigned long request, unsigned long flags)
{
    struct mm_struct *mm = current->mm;
    unsigned long len;
    int ret;
    bool populate;
    LIST_HEAD(uf);

    /* 页对齐长度 */
    len = PAGE_ALIGN(request);
    if (len < request)
        return -ENOMEM;
    if (!len)
        return 0;

    /* 获取mmap_lock */
    if (mmap_write_lock_killable(mm))
        return -EINTR;

    /* 调用do_brk_flags */
    ret = do_brk_flags(addr, len, flags, &uf);
    
    populate = ((mm->def_flags & VM_LOCKED) != 0);
    
    /* 释放锁 */
    mmap_write_unlock(mm);
    userfaultfd_unmap_complete(mm, &uf);
    
    /* 如果需要，预填充页表 */
    if (populate && !ret)
        mm_populate(addr, len);
        
    return ret;
}
EXPORT_SYMBOL(vm_brk_flags);

/* 简化版本 */
int vm_brk(unsigned long addr, unsigned long len)
{
    return vm_brk_flags(addr, len, 0);
}
EXPORT_SYMBOL(vm_brk);
```

### 10.1 RLIMIT_DATA检查

```c
/*
 * include/linux/mm.h
 * 检查数据段资源限制
 */
static inline int check_data_rlimit(unsigned long rlim,
                                    unsigned long new,
                                    unsigned long start,
                                    unsigned long end_data,
                                    unsigned long start_data)
{
    if (rlim < RLIM_INFINITY) {
        /* 检查：(新brk - start_brk) + (end_data - start_data) <= rlim */
        if (((new - start) + (end_data - start_data)) > rlim)
            return -ENOSPC;
    }
    return 0;
}
```

### 10.3 may_expand_vm函数

```c
/*
 * mm/mmap.c
 * 检查是否可以扩展虚拟内存
 */
bool may_expand_vm(struct mm_struct *mm, vm_flags_t flags, unsigned long npages)
{
    /* 检查RLIMIT_AS */
    if (mm->total_vm + npages > rlimit(RLIMIT_AS) >> PAGE_SHIFT)
        return false;

    /* 检查RLIMIT_DATA（仅对数据映射） */
    if (is_data_mapping(flags) &&
        mm->data_vm + npages > rlimit(RLIMIT_DATA) >> PAGE_SHIFT) {
        
        /* Valgrind兼容性处理 */
        if (rlimit(RLIMIT_DATA) == 0 &&
            mm->data_vm + npages <= rlimit_max(RLIMIT_DATA) >> PAGE_SHIFT)
            return true;

        pr_warn_once("%s (%d): VmData %lu exceed data ulimit %lu.\n",
                     current->comm, current->pid,
                     (mm->data_vm + npages) << PAGE_SHIFT,
                     rlimit(RLIMIT_DATA));

        if (!ignore_rlimit_data)
            return false;
    }

    return true;
}
```

## 十一、ELF加载与堆初始化

### 11.1 set_brk函数

```c
/*
 * fs/binfmt_elf.c
 * ELF加载时设置堆的起始位置
 */
static int set_brk(unsigned long start, unsigned long end, int prot)
{
    start = ELF_PAGEALIGN(start);
    end = ELF_PAGEALIGN(end);
    
    if (end > start) {
        /*
         * 映射BSS段的最后部分
         * 如果头文件要求这些页可执行，则遵守（ppc32需要）
         */
        int error = vm_brk_flags(start, end - start,
                prot & PROT_EXEC ? VM_EXEC : 0);
        if (error)
            return error;
    }
    
    /* 设置堆的起始位置 */
    current->mm->start_brk = current->mm->brk = end;
    return 0;
}
```

### 12.2 实现简单的内存分配器

```c
/*
 * 基于sbrk的简单内存分配器
 */
#include <unistd.h>

/* 内存块头 */
struct block_header {
    size_t size;           /* 块大小（不含头） */
    int free;              /* 是否空闲 */
    struct block_header *next;  /* 下一个块 */
};

#define HEADER_SIZE sizeof(struct block_header)
#define ALIGN8(x) (((x) + 7) & ~7)  /* 8字节对齐 */

static struct block_header *head = NULL;

/* 分配内存 */
void *simple_malloc(size_t size)
{
    size = ALIGN8(size);
    
    /* 查找空闲块 */
    struct block_header *curr = head;
    while (curr) {
        if (curr->free && curr->size >= size) {
            curr->free = 0;
            return (void *)(curr + 1);
        }
        curr = curr->next;
    }
    
    /* 没有合适的块，扩展堆 */
    struct block_header *new_block = sbrk(HEADER_SIZE + size);
    if (new_block == (void *)-1)
        return NULL;
    
    new_block->size = size;
    new_block->free = 0;
    new_block->next = NULL;
    
    /* 加入链表 */
    if (!head) {
        head = new_block;
    } else {
        curr = head;
        while (curr->next)
            curr = curr->next;
        curr->next = new_block;
    }
    
    return (void *)(new_block + 1);
}

/* 释放内存 */
void simple_free(void *ptr)
{
    if (!ptr)
        return;
    
    struct block_header *block = (struct block_header *)ptr - 1;
    block->free = 1;
    
    /* 简单实现：不收缩堆 */
    /* 实际实现中，如果尾部是空闲块，可以调用sbrk收缩 */
}
```

## 总结

brk是Linux内核中最古老的堆内存管理机制，虽然现代应用中很少直接使用，但理解其原理对于深入理解Linux内存管理至关重要：

**核心价值**：
1. **简单高效**：线性扩展，一次系统调用分配大块内存
2. **小块内存优化**：glibc malloc对小块内存使用brk
3. **历史兼容**：保持与早期Unix的兼容性

**关键技术**：
- **mm_struct**：存储brk和start_brk边界
- **VMA**：堆区域由vm_area_struct描述
- **资源限制**：RLIMIT_DATA控制堆大小
- **VMA合并**：扩展时尝试合并相邻VMA

**与malloc的关系**：
- 小块内存（< 128KB）：使用brk分配
- 大块内存（> 128KB）：使用mmap分配
- glibc动态调整阈值

**适用场景**：
- 小块、频繁的内存分配
- 需要连续内存的场景
- 长生命周期的对象

**注意事项**：
- 释放中间块无法归还操作系统
- 可能与mmap区域冲突
- 受RLIMIT_DATA限制

### 1.3 IPv4 vs IPv6 对比

| 特性 | IPv4 | IPv6 |
|------|------|------|
| **地址长度** | 32位（4字节） | 128位（16字节） |
| **地址表示** | 点分十进制 | 冒号十六进制 |
| **分片方式** | 路由器可分片 | 只有源节点分片 |
| **头部复杂度** | 可变长度（含选项） | 固定长度（40字节） |
| **广播** | 支持 | 不支持（使用组播） |
| **NAT** | 广泛使用 | 设计上不需要 |
| **路由表** | FIB + dst_entry | FIB6 + rt6_info |

```
│     ┌─────────────────────────────────────────────────────┐    │
│     │ • 可选算法                                           │    │
│     │ • 时间复杂度：O(1) 平均                              │    │
│     │ • 适合少量路由                                       │    │
│     │ • 配置：CONFIG_IP_FIB_HASH                           │    │

│  查找过程：                                                     │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │                                                         │   │

│  │                                                         │   │
```

## 四、dst_entry路由缓存

### 4.1 dst_entry 结构体

```c
/* include/net/dst.h */

/*
 * dst_entry - 目的条目（路由缓存）
 *
 * 协议无关的路由缓存结构，IPv4和IPv6共用
 */
struct dst_entry {
    struct net_device   *dev;           /* 输出设备 */
    struct dst_ops      *ops;           /* 操作函数集 */
    
    unsigned long        _metrics;      /* 路由度量值指针 */
    unsigned long        expires;       /* 过期时间 */
    
#ifdef CONFIG_XFRM
    struct xfrm_state   *xfrm;          /* IPsec状态 */
#endif
    
    /* 输入和输出函数指针 */
    int                  (*input)(struct sk_buff *);
    int                  (*output)(struct net *net, struct sock *sk,
                                   struct sk_buff *skb);
    
    unsigned short       flags;         /* 标志 */
    short                obsolete;      /* 过期标志 */
    
    unsigned short       header_len;    /* 头部预留空间 */
    unsigned short       trailer_len;   /* 尾部预留空间 */
    
    atomic_t             __refcnt;      /* 引用计数 */
    int                  __use;         /* 使用计数 */
    unsigned long        lastuse;       /* 最后使用时间 */
    
    struct lwtunnel_state *lwtstate;    /* 载道状态 */
    struct rcu_head       rcu_head;     /* RCU回调头 */
    
    short                error;         /* 错误码 */
    __u32                tclassid;      /* 流类别ID */
};
```

### 4.2 rtable 结构体（IPv4路由缓存）

```c
/* include/net/route.h */

/*
 * rtable - IPv4路由缓存条目
 *
 * 继承自dst_entry，添加IPv4特定信息
 */
struct rtable {
    struct dst_entry    dst;            /* 基础目的条目 */
    
    int                 rt_genid;       /* 生成ID */
    unsigned int        rt_flags;       /* 路由标志 */
    __u16               rt_type;        /* 路由类型 */
    __u8                rt_is_input;    /* 是否为输入路由 */
    __u8                rt_uses_gateway;/* 是否使用网关 */
    
    int                 rt_iif;         /* 输入接口索引 */
    
    u8                  rt_gw_family;   /* 网关协议族 */
    
    /* 网关地址 */
    union {
        __be32          rt_gw4;         /* IPv4网关 */
        struct in6_addr rt_gw6;         /* IPv6网关 */
    };
    
    /* MTU相关 */
    u32                 rt_mtu_locked:1,
                        rt_pmtu:31;
    
    struct list_head    rt_uncached;    /* 未缓存链表 */
```

## 五、路由查找流程

### 5.1 ip_route_input 流程

```c
/* net/ipv4/route.c */

/*
 * ip_route_input_noref - 输入路由查找（不增加引用计数）
 * @skb: 数据包
 * @daddr: 目的地址
 * @saddr: 源地址
 * @tos: 服务类型
 * @dev: 接收设备
 */
int ip_route_input_noref(struct sk_buff *skb, __be32 daddr, __be32 saddr,
             u8 tos, struct net_device *dev)
{
    struct fib_result res;
    int err;

    tos &= IPTOS_RT_MASK;
    rcu_read_lock();
    err = ip_route_input_rcu(skb, daddr, saddr, tos, dev, &res);
    rcu_read_unlock();

    return err;
}
EXPORT_SYMBOL(ip_route_input_noref);
```

### 5.2 ip_route_output 流程

```c
/* net/ipv4/route.c */

/*
 * ip_route_output_key_hash - 输出路由查找
 * @net: 网络命名空间
 * @fl4: 流标识
 * @skb: 数据包（可选）
 */
struct rtable *ip_route_output_key_hash(struct net *net, struct flowi4 *fl4,
                    const struct sk_buff *skb)
{
    struct fib_result res = {
        .type       = RTN_UNSPEC,
        .fi         = NULL,
        .table      = NULL,
        .tclassid   = 0,
    };
    struct rtable *rth;

    fl4->flowi4_iif = LOOPBACK_IFINDEX;
    ip_rt_fix_tos(fl4);

    rcu_read_lock();
    rth = ip_route_output_key_hash_rcu(net, fl4, &res, skb);
    rcu_read_unlock();

    return rth;
}
EXPORT_SYMBOL_GPL(ip_route_output_key_hash);
```

### 7.2 ip_defrag 函数

```c
/* net/ipv4/ip_fragment.c */

/*
 * ip_defrag - IP分片重组
 * @net: 网络命名空间
 * @skb: 接收到的IP分片
 * @user: 用户标识
 *
 * 处理接收到的IP分片，进行重组
 */
int ip_defrag(struct net *net, struct sk_buff *skb, u32 user)
{
    struct net_device *dev = skb->dev ? : skb_dst(skb)->dev;
    int vif = l3mdev_master_ifindex_rcu(dev);
    struct ipq *qp;

    __IP_INC_STATS(net, IPSTATS_MIB_REASMREQDS);

    /* 查找或创建分片队列 */
    qp = ip_find(net, ip_hdr(skb), user, vif);
    if (qp) {
        int ret;

        spin_lock(&qp->q.lock);

        /* 将分片加入队列 */
        ret = ip_frag_queue(qp, skb);

        spin_unlock(&qp->q.lock);
        ipq_put(qp);
        return ret;
    }

    __IP_INC_STATS(net, IPSTATS_MIB_REASMFAILS);
    kfree_skb(skb);
    return -ENOMEM;
}
EXPORT_SYMBOL(ip_defrag);
```

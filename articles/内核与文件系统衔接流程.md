### 1.2 main() 函数中的初始化顺序

定义在 **`init/main.c`**：

```c
void main(void)
{
    /* 1. 获取硬件参数 */
    ROOT_DEV = ORIG_ROOT_DEV;           /* 根设备号 */
    drive_info = DRIVE_INFO;            /* 硬盘参数 */
    
    /* 2. 计算内存布局 */
    memory_end = (1 << 20) + (EXT_MEM_K << 10);
    ...
    
    /* 3. 初始化内存管理 */
    mem_init(main_memory_start, memory_end);
    
    /* 4. 初始化中断处理 */
    trap_init();
    
    /* 5. 初始化块设备 */
    blk_dev_init();
    
    /* 6. 初始化字符设备 */
    chr_dev_init();
    tty_init();
    
    /* 7. 初始化调度器 */
    sched_init();
    
    /* 8. 初始化文件系统缓冲区 */
    buffer_init(buffer_memory_end);
    
    /* 9. 初始化硬盘和软驱 */
    hd_init();
    floppy_init();
    
    /* 10. 开中断 */
    sti();
    
    /* 11. 切换到用户态 */
    move_to_user_mode();
    
    /* 12. 创建 init 进程 */
    if (!fork()) {
        init();
    }
    
    /* 13. 进程 0 空闲循环 */
    for (;;)
        pause();
}
```

## 2. init 进程与用户空间启动

### 2.1 init() 函数详解

```c
void init(void)
{
    int pid, i;

    /* 1. 安装根文件系统 */
    setup((void *) &drive_info);
    
    /* 2. 打开控制台 */
    (void) open("/dev/tty0", O_RDWR, 0);   /* stdin  */
    (void) dup(0);                          /* stdout */
    (void) dup(0);                          /* stderr */
    
    /* 3. 打印系统信息 */
    printf("%d buffers = %d bytes buffer space\n\r", 
           NR_BUFFERS, NR_BUFFERS * BLOCK_SIZE);
    printf("Free mem: %d bytes\n\r", 
           memory_end - main_memory_start);
    
    /* 4. 执行启动脚本 */
    if (!(pid = fork())) {
        close(0);
        if (open("/etc/rc", O_RDONLY, 0))
            _exit(1);
        execve("/bin/sh", argv_rc, envp_rc);
        _exit(2);
    }
    
    /* 5. 等待启动脚本完成 */
    if (pid > 0)
        while (pid != wait(&i))
            ;
    
    /* 6. 启动登录 shell */
    while (1) {
        if ((pid = fork()) < 0) {
            printf("Fork failed in init\r\n");
            continue;
        }
        if (!pid) {
            close(0); close(1); close(2);
            setsid();
            (void) open("/dev/tty0", O_RDWR, 0);
            (void) dup(0);
            (void) dup(0);
            _exit(execve("/bin/sh", argv, envp));
        }
        
        /* 7. 等待 shell 退出，然后重启 */
        while (1)
            if (pid == wait(&i))
                break;
        printf("\n\rchild %d died with code %04x\n\r", pid, i);
        sync();
    }
    _exit(0);
}
```

### 2.3 setup() 系统调用

setup() 是内核与文件系统衔接的关键：

```c
/* 在 hd.c 中实现 */
int sys_setup(void * BIOS)
{
    static int callable = 1;
    int i, drive;
    struct partition *p;
    struct buffer_head *bh;

    if (!callable)
        return -1;
    callable = 0;
    
    /* 1. 读取 BIOS 中的硬盘参数 */
    for (drive = 0; drive < 2; drive++) {
        hd_info[drive].cyl = *(unsigned short *) BIOS;
        hd_info[drive].head = *(unsigned char *) (2 + BIOS);
        ...
    }
    
    /* 2. 读取分区表 */
    for (drive = 0; drive < NR_HD; drive++) {
        bh = bread(0x300 + drive * 5, 0);  /* 读取 MBR */
        
        /* 验证分区表签名 */
        if (bh->b_data[510] != 0x55 || 
            (unsigned char)bh->b_data[511] != 0xAA) {
            panic("Bad partition table");
        }
        
        /* 解析分区表 */
        p = 0x1BE + (void *)bh->b_data;
        for (i = 1; i < 5; i++, p++) {
            hd[i + 5 * drive].start_sect = p->start_sect;
            hd[i + 5 * drive].nr_sects = p->nr_sects;
        }
        brelse(bh);
    }
    
    /* 3. 尝试加载 RAM 盘 */
    rd_load();
    
    /* 4. 挂载根文件系统 */
    mount_root();
    
    return 0;
}
```

## 3. 文件系统初始化流程

### 3.1 mount_root() 函数

定义在 **`fs/super.c`**：

```c
void mount_root(void)
{
    int i, free;
    struct super_block *p;
    struct m_inode *mi;

    /* 1. 验证 inode 结构大小 */
    if (32 != sizeof(struct d_inode))
        panic("bad i-node size");
    
    /* 2. 初始化文件表 */
    for (i = 0; i < NR_FILE; i++)
        file_table[i].f_count = 0;
    
    /* 3. 如果是软盘，提示用户插入 */
    if (MAJOR(ROOT_DEV) == 2) {
        printk("Insert root floppy and press ENTER");
        wait_for_keypress();
    }
    
    /* 4. 初始化超级块数组 */
    for (p = &super_block[0]; p < &super_block[NR_SUPER]; p++) {
        p->s_dev = 0;
        p->s_lock = 0;
        p->s_wait = NULL;
    }
    
    /* 5. 读取根文件系统超级块 */
    if (!(p = read_super(ROOT_DEV)))
        panic("Unable to mount root");
    
    /* 6. 读取根 inode */
    if (!(mi = iget(ROOT_DEV, ROOT_INO)))
        panic("Unable to read root i-node");
    
    mi->i_count += 3;
    p->s_isup = p->s_imount = mi;
    
    /* 7. 设置当前进程的根目录和当前目录 */
    current->pwd = mi;
    current->root = mi;
    
    /* 8. 统计空闲块和空闲 inode */
    free = 0;
    i = p->s_nzones;
    while (--i >= 0)
        if (!set_bit(i & 8191, p->s_zmap[i >> 13]->b_data))
            free++;
    printk("%d/%d free blocks\n\r", free, p->s_nzones);
    
    free = 0;
    i = p->s_ninodes + 1;
    while (--i >= 0)
        if (!set_bit(i & 8191, p->s_imap[i >> 13]->b_data))
            free++;
    printk("%d/%d free inodes\n\r", free, p->s_ninodes);
}
```

### 3.3 RAM 盘加载

定义在 **`kernel/blk_drv/ramdisk.c`**：

```c
void rd_load(void)
{
    struct buffer_head *bh;
    struct super_block s;
    int block = 256;
    int i = 1;
    int nblocks;
    char *cp;
    
    /* 1. 检查是否配置了 RAM 盘 */
    if (!rd_length)
        return;
    
    printk("Ram disk: %d bytes, starting at 0x%x\n", 
           rd_length, (int)rd_start);
    
    /* 2. 检查根设备是否为软盘 */
    if (MAJOR(ROOT_DEV) != 2)
        return;
    
    /* 3. 读取软盘上的文件系统超级块 */
    bh = breada(ROOT_DEV, block + 1, block, block + 2, -1);
    if (!bh) {
        printk("Disk error while looking for ramdisk!\n");
        return;
    }
    
    *((struct d_super_block *)&s) = 
        *((struct d_super_block *)bh->b_data);
    brelse(bh);
    
    /* 4. 验证是否为有效的文件系统 */
    if (s.s_magic != SUPER_MAGIC)
        return;  /* 没有 RAM 盘镜像，正常软盘启动 */
    
    /* 5. 检查 RAM 盘大小 */
    nblocks = s.s_nzones << s.s_log_zone_size;
    if (nblocks > (rd_length >> BLOCK_SIZE_BITS)) {
        printk("Ram disk image too big!\n");
        return;
    }
    
    /* 6. 加载 RAM 盘镜像 */
    printk("Loading %d bytes into ram disk... 0000k", 
           nblocks << BLOCK_SIZE_BITS);
    
    cp = rd_start;
    while (nblocks) {
        bh = bread(ROOT_DEV, block);
        if (!bh) {
            printk("I/O error on block %d\n", block);
            return;
        }
        memcpy(cp, bh->b_data, BLOCK_SIZE);
        brelse(bh);
        printk("\010\010\010\010\010%4dk", i);
        cp += BLOCK_SIZE;
        block++;
        nblocks--;
        i++;
    }
    
    printk("\010\010\010\010\010done \n");
    
    /* 7. 切换根设备为 RAM 盘 */
    ROOT_DEV = 0x0101;  /* /dev/ram */
}
```

## 4. 根文件系统切换

### 4.1 Linux 的 pivot_root 机制

现代 Linux 使用 pivot_root 切换根文件系统：

```c
/* pivot_root 系统调用 */
int pivot_root(const char *new_root, const char *put_old);

/* 使用示例 */
/* 1. 挂载新的根文件系统 */
mount("/dev/sda1", "/newroot", "ext4", 0, NULL);

/* 2. 切换根文件系统 */
pivot_root("/newroot", "/newroot/oldroot");

/* 3. 卸载旧的根文件系统 */
chdir("/");
umount("/oldroot");
```

### 4.3 initramfs 的 init 脚本

```bash
#!/bin/sh

# 1. 挂载必要的虚拟文件系统
mount -t proc proc /proc
mount -t sysfs sysfs /sys
mount -t devtmpfs devtmpfs /dev

# 2. 加载必要的内核模块
modprobe ext4
modprobe sd_mod

# 3. 等待设备就绪
sleep 1

# 4. 挂载真实根文件系统
mount -t ext4 /dev/sda1 /mnt

# 5. 切换到真实根文件系统
exec switch_root /mnt /sbin/init
```

## 6. 总结

### 6.1 关键衔接点

| 衔接点 | 说明 |
|-------|------|
| ROOT_DEV | 内核启动时确定的根设备号 |
| setup() | 初始化硬盘并挂载根文件系统 |
| mount_root() | 读取超级块和根 inode |
| init 进程 | 第一个用户空间进程 |
| /dev/console | 控制台设备，init 打开后成为标准 I/O |
| /bin/sh | 默认的 Shell 程序 |

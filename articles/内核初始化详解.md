## 概述

`main()` 函数是 Linux 0.11 内核初始化的入口点，由 `head.s` 在完成基本的硬件设置后调用。`main()` 函数负责初始化内核的各个子系统，创建第一个用户进程，并最终让系统进入正常运行状态。

## main 函数

### 文件位置

**定义位置**：**`init/main.c:107-139`**

### 完整代码

```c
void main(void)		/* This really IS void, no error here. */
{			/* The startup routine assumes (well, ...) this */
/*
 * Interrupts are still disabled. Do necessary setups, then
 * enable them
 */
 	ROOT_DEV = ORIG_ROOT_DEV;
 	drive_info = DRIVE_INFO;
	memory_end = (1<<20) + (EXT_MEM_K<<10);
	memory_end &= 0xfffff000;
	if (memory_end > 16*1024*1024)
		memory_end = 16*1024*1024;
	if (memory_end > 12*1024*1024) 
		buffer_memory_end = 4*1024*1024;
	else if (memory_end > 6*1024*1024)
		buffer_memory_end = 2*1024*1024;
	else
		buffer_memory_end = 1*1024*1024;
	main_memory_start = buffer_memory_end;
#ifdef RAMDISK
	main_memory_start += rd_init(main_memory_start, RAMDISK*1024);
#endif
	mem_init(main_memory_start,memory_end);
	trap_init();
	blk_dev_init();
	chr_dev_init();
	tty_init();
	time_init();
	sched_init();
	buffer_init(buffer_memory_end);
	hd_init();
	floppy_init();
	sti();
	move_to_user_mode();
	if (!fork()) {		/* we count on this going ok */
		init();
	}
	for(;;) pause();
}
```

### 硬件参数获取

main 函数首先从内存 0x90000 处读取 setup.s 保存的硬件参数：

```c
#define EXT_MEM_K (*(unsigned short *)0x90002)    /* 扩展内存大小 (KB) */
#define DRIVE_INFO (*(struct drive_info *)0x90080) /* 硬盘参数表 */
#define ORIG_ROOT_DEV (*(unsigned short *)0x901FC) /* 根设备号 */
```

**参数来源**：

| 地址 | 内容 | 说明 |
|------|------|------|
| 0x90002 | EXT_MEM_K | 扩展内存大小，单位 KB |
| 0x90080 | DRIVE_INFO | 硬盘参数表 (32字节) |
| 0x901FC | ORIG_ROOT_DEV | 根文件系统设备号 |

### 内存布局计算

```c
memory_end = (1<<20) + (EXT_MEM_K<<10);  /* 基础 1MB + 扩展内存 */
memory_end &= 0xfffff000;                 /* 4KB 对齐 */
if (memory_end > 16*1024*1024)
    memory_end = 16*1024*1024;            /* 最大 16MB */
```

**缓冲区边界确定**：

```c

else if (memory_end > 6*1024*1024)
    buffer_memory_end = 2*1024*1024;      /* 6-12MB 内存: 2MB 缓冲区 */
else
    buffer_memory_end = 1*1024*1024;      /* <6MB 内存: 1MB 缓冲区 */
```

## 内存管理初始化 mem_init

### 文件位置

**定义位置**：**`mm/memory.c:336-353`**

### 功能说明

`mem_init()` 初始化物理内存页面管理，设置每个页面的使用计数。

### 代码实现

```c
#define LOW_MEM 0x100000                    /* 主内存起始地址 1MB */
#define PAGING_MEMORY (15*1024*1024)        /* 可分页内存大小 */
#define PAGING_PAGES (PAGING_MEMORY>>12)    /* 页面数量 (4KB/页) */
#define MAP_NR(addr) (((addr)-LOW_MEM)>>12) /* 地址到页面号映射 */
#define USED 100                            /* 已使用标记 */

static unsigned char mem_map [ PAGING_PAGES ] = {0,};  /* 页面使用计数 */

void mem_init(long start_mem, long end_mem)
{
    int i;

    HIGH_MEMORY = end_mem;                  /* 记录内存结束地址 */
    for (i=0 ; i<PAGING_PAGES ; i++)
        mem_map[i] = USED;                  /* 标记所有页面为已使用 */
    i = MAP_NR(start_mem);                  /* 主内存起始页面号 */
    end_mem -= start_mem;                   /* 计算主内存大小 */
    end_mem >>= 12;                         /* 转换为页面数 */
    while (end_mem-->0)
        mem_map[i++]=0;                     /* 标记可用页面为 0 */
}
```

### mem_map 数组

```
mem_map 数组结构:
┌─────────────────────────────────────────────────────────────────────────┐
│ 索引    │ 内容 │ 说明                                                  │
├─────────────────────────────────────────────────────────────────────────┤
│ 0       │ 100  │ 对应 LOW_MEM (1MB) 以下的内存，标记为已使用            │
│ 1       │ 100  │ 内核代码段，不可分配                                  │
│ ...     │ 100  │ 内核数据段、页表等                                    │
├─────────────────────────────────────────────────────────────────────────┤
│ i       │ 0    │ 主内存起始页面，可分配                                │
│ i+1     │ 0    │ 可分配                                                │
│ ...     │ 0    │ 可分配                                                │
│ end-1   │ 0    │ 可分配                                                │
└─────────────────────────────────────────────────────────────────────────┘

mem_map[i] 的含义:
- 0: 页面空闲，可分配
- 1: 页面被使用 1 次
- n: 页面被共享使用 n 次
- 100 (USED): 页面保留，不可分配 (内核使用)
```

### 页面分配函数 get_free_page

```c
unsigned long get_free_page(void)
{
register unsigned long __res asm("ax");

__asm__("std ; repne ; scasb\n\t"      /* 反向搜索 mem_map 找空闲页 */
    "jne 1f\n\t"
    "movb $1,1(%%edi)\n\t"             /* 标记为已使用 */
    "sall $12,%%ecx\n\t"               /* 页号左移12位 = 物理地址 */
    "addl %2,%%ecx\n\t"                /* 加上 LOW_MEM */
    "movl %%ecx,%%edx\n\t"
    "movl $1024,%%ecx\n\t"             /* 清零页面 */
    "leal 4092(%%edx),%%edi\n\t"
    "rep ; stosl\n\t"
    "movl %%edx,%%eax\n"               /* 返回物理地址 */
    "1:"
    :"=a" (__res)
    :"0" (0),"i" (LOW_MEM),"c" (PAGING_PAGES),
    "D" (mem_map+PAGING_PAGES-1)
    :"di","cx","dx");
return __res;
}
```

## 中断处理初始化 trap_init

### 文件位置

**定义位置**：**`kernel/traps.c:182-208`**

### 功能说明

`trap_init()` 初始化中断描述符表 (IDT)，设置各种异常和中断的处理门。

### 代码实现

```c
void trap_init(void)
{
    int i;

    set_trap_gate(0, &divide_error);          /* 除零错误 */
    set_trap_gate(1, &debug);                 /* 调试异常 */
    set_trap_gate(2, &nmi);                   /* 不可屏蔽中断 */
    set_system_gate(3, &int3);                /* 断点指令 */
    set_system_gate(4, &overflow);            /* 溢出 */
    set_system_gate(5, &bounds);              /* 边界检查 */
    set_trap_gate(6, &invalid_op);            /* 无效操作码 */
    set_trap_gate(7, &device_not_available);  /* 设备不可用 */
    set_trap_gate(8, &double_fault);          /* 双重错误 */
    set_trap_gate(9, &coprocessor_segment_overrun); /* 协处理器段越界 */
    set_trap_gate(10, &invalid_TSS);          /* 无效 TSS */
    set_trap_gate(11, &segment_not_present);  /* 段不存在 */
    set_trap_gate(12, &stack_segment);        /* 栈段错误 */
    set_trap_gate(13, &general_protection);   /* 一般保护错误 */
    set_trap_gate(14, &page_fault);           /* 页错误 */
    set_trap_gate(15, &reserved);             /* 保留 */
    set_trap_gate(16, &coprocessor_error);    /* 协处理器错误 */
    for (i=17; i<48; i++)
        set_trap_gate(i, &reserved);          /* 保留中断号 */
    set_trap_gate(45, &irq13);                /* 协处理器中断 */
    outb_p(inb_p(0x21)&0xfb, 0x21);           /* 允许 IRQ2 (级联) */
    outb(inb_p(0xA1)&0xdf, 0xA1);             /* 允许 IRQ13 (协处理器) */
    set_trap_gate(39, &parallel_interrupt);   /* 并口中断 */
}
```

### 中断门设置函数

**定义位置**：**`include/asm/system.h:33-44`**

```c
#define _set_gate(gate_addr,type,dpl,addr) \
__asm__ ("movw %%dx,%%ax\n\t" \
    "movw %0,%%dx\n\t" \
    "movl %%eax,%1\n\t" \
    "movl %%edx,%2" \
    : \
    : "i" ((short) (0x8000+(dpl<<13)+(type<<8))), \
    "o" (*((char *) (gate_addr))), \
    "o" (*(4+(char *) (gate_addr))), \
    "d" ((char *) (addr)),"a" (0x00080000))

#define set_intr_gate(n,addr) \
    _set_gate(&idt[n],14,0,addr)    /* DPL=0, 类型=中断门 */

#define set_trap_gate(n,addr) \
    _set_gate(&idt[n],15,0,addr)    /* DPL=0, 类型=陷阱门 */

#define set_system_gate(n,addr) \
    _set_gate(&idt[n],15,3,addr)    /* DPL=3, 类型=陷阱门 */
```

## 块设备初始化 blk_dev_init

### 文件位置

**定义位置**：**`kernel/blk_drv/ll_rw_blk.c:153-162`**

### 功能说明

`blk_dev_init()` 初始化块设备请求队列，将所有请求项标记为空闲。

### 代码实现

```c
#define NR_REQUEST 32    /* 请求数量 */

struct request {
    int dev;                    /* 设备号 (-1 表示空闲) */
    int cmd;                    /* READ 或 WRITE */
    int errors;                 /* 错误计数 */
    unsigned long sector;       /* 起始扇区 */
    unsigned long nr_sectors;   /* 扇区数 */
    char * buffer;              /* 数据缓冲区 */
    struct task_struct * waiting; /* 等待进程 */
    struct buffer_head * bh;    /* 缓冲区头 */
    struct request * next;      /* 下一个请求 */
};

struct request request[NR_REQUEST];

void blk_dev_init(void)
{
    int i;

    for (i=0 ; i<NR_REQUEST ; i++) {
        request[i].dev = -1;      /* -1 表示请求项空闲 */
        request[i].next = NULL;
    }
}
```

### 块设备结构

```c
#define NR_BLK_DEV 7    /* 块设备数量 */

struct blk_dev_struct {
    void (*request_fn)(void);       /* 请求处理函数 */
    struct request * current_request; /* 当前请求 */
};

struct blk_dev_struct blk_dev[NR_BLK_DEV] = {
    { NULL, NULL },     /* 0: no_dev */
    { NULL, NULL },     /* 1: dev mem (虚拟盘) */
    { NULL, NULL },     /* 2: dev fd (软驱) */
    { NULL, NULL },     /* 3: dev hd (硬盘) */
    { NULL, NULL },     /* 4: dev ttyx */
    { NULL, NULL },     /* 5: dev tty */
    { NULL, NULL }      /* 6: dev lp */
};
```

## 时间初始化 time_init

### 文件位置

**定义位置**：**`init/main.c:49-66`**

### 功能说明

`time_init()` 从 CMOS 读取实时时钟，设置系统启动时间。

### 代码实现

```c
#define CMOS_READ(addr) ({ \
outb_p(0x80|addr,0x70); \
inb_p(0x71); \
})

#define BCD_TO_BIN(val) ((val)=((val)&15) + ((val)>>4)*10)

static void time_init(void)
{
    struct tm time;

    do {
        time.tm_sec = CMOS_READ(0);     /* 秒 */
        time.tm_min = CMOS_READ(2);     /* 分 */
        time.tm_hour = CMOS_READ(4);    /* 时 */
        time.tm_mday = CMOS_READ(7);    /* 日 */
        time.tm_mon = CMOS_READ(8);     /* 月 */
        time.tm_year = CMOS_READ(9);    /* 年 */
    } while (time.tm_sec != CMOS_READ(0));  /* 确保读取一致 */
    
    BCD_TO_BIN(time.tm_sec);            /* BCD 转二进制 */
    BCD_TO_BIN(time.tm_min);
    BCD_TO_BIN(time.tm_hour);
    BCD_TO_BIN(time.tm_mday);
    BCD_TO_BIN(time.tm_mon);
    BCD_TO_BIN(time.tm_year);
    time.tm_mon--;                      /* 月份从 0 开始 */
    startup_time = kernel_mktime(&time); /* 转换为 Unix 时间戳 */
}
```

## 进程调度初始化 sched_init

### 文件位置

**定义位置**：**`kernel/sched.c:385-412`**

### 功能说明

`sched_init()` 初始化进程调度系统，设置进程 0 的 TSS 和 LDT，初始化定时器中断。

### 代码实现

```c
void sched_init(void)
{
    int i;
    struct desc_struct * p;

    if (sizeof(struct sigaction) != 16)
        panic("Struct sigaction MUST be 16 bytes");
    
    /* 设置进程 0 的 TSS 和 LDT 描述符 */
    set_tss_desc(gdt+FIRST_TSS_ENTRY, &(init_task.task.tss));
    set_ldt_desc(gdt+FIRST_LDT_ENTRY, &(init_task.task.ldt));
    
    /* 清空其他进程的 TSS 和 LDT 描述符 */
    p = gdt+2+FIRST_TSS_ENTRY;
    for(i=1; i<NR_TASKS; i++) {
        task[i] = NULL;
        p->a = p->b = 0;      /* TSS */
        p++;
        p->a = p->b = 0;      /* LDT */
        p++;
    }
    
    /* 清除 NT 标志 (嵌套任务) */
    __asm__("pushfl ; andl $0xffffbfff,(%esp) ; popfl");
    
    /* 加载任务寄存器和 LDT */
    ltr(0);    /* 加载 TSS 选择子 */
    lldt(0);   /* 加载 LDT 选择子 */
    
    /* 初始化 8253 定时器 (100Hz) */
    outb_p(0x36, 0x43);              /* 模式 3, 方波发生器 */
    outb_p(LATCH & 0xff, 0x40);      /* 低字节 */
    outb(LATCH >> 8, 0x40);          /* 高字节 */
    
    /* 设置定时器中断门 */
    set_intr_gate(0x20, &timer_interrupt);
    
    /* 允许定时器中断 (IRQ0) */
    outb(inb_p(0x21)&~0x01, 0x21);
    
    /* 设置系统调用门 */
    set_system_gate(0x80, &system_call);
}
```

## 缓冲区初始化 buffer_init

### 文件位置

**定义位置**：**`fs/buffer.c:351-381`**

### 功能说明

`buffer_init()` 初始化文件系统缓冲区，建立缓冲区链表和哈希表。

### 代码实现

```c
extern int end;    /* 内核结束地址 (由链接器定义) */
struct buffer_head * start_buffer = (struct buffer_head *) &end;
struct buffer_head * hash_table[NR_HASH];
static struct buffer_head * free_list;
int NR_BUFFERS = 0;

void buffer_init(long buffer_end)
{
    struct buffer_head * h = start_buffer;
    void * b;
    int i;

    if (buffer_end == 1<<20)
        b = (void *) (640*1024);    /* 1MB 边界，跳过显存区 */
    else
        b = (void *) buffer_end;
    
    /* 从高地址向低地址分配缓冲区 */
    while ( (b -= BLOCK_SIZE) >= ((void *) (h+1)) ) {
        h->b_dev = 0;               /* 设备号 */
        h->b_dirt = 0;              /* 脏标志 */
        h->b_count = 0;             /* 引用计数 */
        h->b_lock = 0;              /* 锁标志 */
        h->b_uptodate = 0;          /* 更新标志 */
        h->b_wait = NULL;           /* 等待队列 */
        h->b_next = NULL;           /* 哈希链表下一个 */
        h->b_prev = NULL;           /* 哈希链表上一个 */
        h->b_data = (char *) b;     /* 数据指针 */
        h->b_prev_free = h-1;       /* 空闲链表上一个 */
        h->b_next_free = h+1;       /* 空闲链表下一个 */
        h++;
        NR_BUFFERS++;
        if (b == (void *) 0x100000) /* 跳过显存区 */
            b = (void *) 0xA0000;
    }
    h--;
    free_list = start_buffer;       /* 空闲链表头 */
    free_list->b_prev_free = h;     /* 循环链表 */
    h->b_next_free = free_list;
    
    /* 初始化哈希表 */
    for (i=0; i<NR_HASH; i++)
        hash_table[i] = NULL;
}
```

## 切换到用户态 move_to_user_mode

### 文件位置

**定义位置**：**`include/asm/system.h:1-12`**

### 功能说明

`move_to_user_mode()` 将进程 0 从内核态切换到用户态。这是通过模拟中断返回 (IRET) 实现的。

### 代码实现

```c
#define move_to_user_mode() \
__asm__ ("movl %%esp,%%eax\n\t"    /* 保存当前栈指针 */ \
    "pushl $0x17\n\t"              /* SS = 用户数据段选择子 */ \
    "pushl %%eax\n\t"              /* ESP = 用户栈指针 */ \
    "pushfl\n\t"                   /* EFLAGS */ \
    "pushl $0x0f\n\t"              /* CS = 用户代码段选择子 */ \
    "pushl $1f\n\t"                /* EIP = 返回地址 (标号 1) */ \
    "iret\n"                       /* 中断返回，切换到用户态 */ \
    "1:\tmovl $0x17,%%eax\n\t"     /* 设置用户数据段 */ \
    "movw %%ax,%%ds\n\t"           /* DS = 用户数据段 */ \
    "movw %%ax,%%es\n\t"           /* ES = 用户数据段 */ \
    "movw %%ax,%%fs\n\t"           /* FS = 用户数据段 */ \
    "movw %%ax,%%gs"               /* GS = 用户数据段 */ \
    :::"ax")
```

## 创建第一个进程 fork

### 文件位置

**定义位置**：**`init/main.c`**

### 功能说明

main 函数通过 `fork()` 创建进程 1 (init 进程)。

### 代码实现

```c
static inline _syscall0(int, fork)    /* 内联 fork 系统调用 */

void main(void)
{
    /* ... 初始化代码 ... */
    
    sti();
    move_to_user_mode();
    
    if (!fork()) {        /* fork 返回 0 表示子进程 */
        init();           /* 子进程执行 init() */
    }
    
    for(;;) pause();      /* 父进程 (进程 0) 进入空闲循环 */
}
```

### fork 系统调用

```c
#define _syscall0(type, name) \
type name(void) \
{ \
    long __res; \
    __asm__ volatile ("int $0x80" \
        : "=a" (__res) \
        : "0" (__NR_##name)); \
    if (__res >= 0) \
        return (type) __res; \
    errno = -__res; \
    return -1; \
}

/* fork() 展开后:
 * int fork(void) {
 *     long __res;
 *     __asm__ volatile ("int $0x80" : "=a" (__res) : "0" (__NR_fork));
 *     if (__res >= 0) return (int) __res;
 *     errno = -__res;
 *     return -1;
 * }
 */
```

## init 进程

### 文件位置

**定义位置**：**`init/main.c:156-202`**

### 功能说明

`init()` 是进程 1 的主函数，负责：
1. 安装根文件系统
2. 打开控制台设备
3. 创建 shell 进程
4. 回收孤儿进程

### 代码实现

```c
static char * argv_rc[] = { "/bin/sh", NULL };
static char * envp_rc[] = { "HOME=/", NULL };

static char * argv[] = { "-/bin/sh", NULL };
static char * envp[] = { "HOME=/usr/root", NULL };

void init(void)
{
    int pid, i;

    /* 安装根文件系统 */
    setup((void *) &drive_info);
    
    /* 打开控制台 */
    (void) open("/dev/tty0", O_RDWR, 0);   /* stdin */
    (void) dup(0);                          /* stdout */
    (void) dup(0);                          /* stderr */
    
    /* 打印系统信息 */
    printf("%d buffers = %d bytes buffer space\n\r", 
        NR_BUFFERS, NR_BUFFERS*BLOCK_SIZE);
    printf("Free mem: %d bytes\n\r", 
        memory_end - main_memory_start);
    
    /* 执行 /etc/rc 脚本 */
    if (!(pid = fork())) {
        close(0);
        if (open("/etc/rc", O_RDONLY, 0))
            _exit(1);
        execve("/bin/sh", argv_rc, envp_rc);
        _exit(2);
    }
    if (pid > 0)
        while (pid != wait(&i))
            /* 等待 rc 脚本执行完成 */;
    
    /* 创建登录 shell */
    while (1) {
        if ((pid = fork()) < 0) {
            printf("Fork failed in init\r\n");
            continue;
        }
        if (!pid) {
            /* 子进程执行 shell */
            close(0); close(1); close(2);
            setsid();
            (void) open("/dev/tty0", O_RDWR, 0);
            (void) dup(0);
            (void) dup(0);
            _exit(execve("/bin/sh", argv, envp));
        }
        /* 父进程等待子进程退出 */
        while (1)
            if (pid == wait(&i))
                break;
        printf("\n\rchild %d died with code %04x\n\r", pid, i);
        sync();
    }
    _exit(0);
}
```

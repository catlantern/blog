```

│                二层交换 vs 三层路由                           │

│                                                             │

│  ┌─────────────────────────────────────────────────────┐   │

│  │  转发依据：MAC地址                                      │   │

│  ┌─────────────────────────────────────────────────────┐   │
```

## 三、Bridge概述

### 3.1 什么是Bridge

**Bridge（网桥）** 是Linux内核实现的软件以太网交换机，用于连接多个网络设备，实现二层转发功能。

简单来说，Bridge就是**软件实现的交换机**：
- 连接多个网络设备（物理网卡、虚拟网卡、veth等）
- 基于MAC地址转发数据包
- 自动学习MAC地址表
- 支持STP防止环路
- 支持VLAN隔离

## 五、br_handle_frame()转发逻辑

### 5.1 br_handle_frame()函数

```c
/* net/bridge/br_input.c */

/*
 * br_handle_frame - Bridge接收数据包的入口函数
 * 
 * 这是Bridge处理数据包的第一步
 * 每个连接到Bridge的设备都会注册这个rx_handler
 */
static rx_handler_result_t br_handle_frame(struct sk_buff **pskb)
{
    struct net_bridge_port *p;
    struct sk_buff *skb = *pskb;
    const unsigned char *dest = eth_hdr(skb)->h_dest;

    /* 1. 检查是否为环回包 */
    if (unlikely(skb->pkt_type == PACKET_LOOPBACK))
        return RX_HANDLER_PASS;

    /* 2. 检查源MAC地址有效性 */
    if (!is_valid_ether_addr(eth_hdr(skb)->h_source))
        goto drop;

    /* 3. 共享检查（确保skb可修改） */
    skb = skb_share_check(skb, GFP_ATOMIC);
    if (!skb)
        return RX_HANDLER_CONSUMED;

    /* 4. 清空控制块 */
    memset(skb->cb, 0, sizeof(struct br_input_skb_cb));

    /* 5. 获取Bridge端口 */
    p = br_port_get_rcu(skb->dev);
    
    /* 6. 处理VLAN隧道 */
    if (p->flags & BR_VLAN_TUNNEL)
        br_handle_ingress_vlan_tunnel(skb, p, nbp_vlan_group_rcu(p));

    /* 7. 处理链路本地地址（01:80:C2:xx:xx:xx） */
    if (unlikely(is_link_local_ether_addr(dest))) {
        u16 fwd_mask = p->br->group_fwd_mask_required;

        fwd_mask |= p->group_fwd_mask;
        
        switch (dest[5]) {
        case 0x00:  /* Bridge Group Address（STP） */
            /* 如果STP关闭或允许转发 */
            if (p->br->stp_enabled == BR_NO_STP ||
                fwd_mask & (1u << dest[5]))
                goto forward;
            
            /* 传递给本地协议栈处理STP */
            *pskb = skb;
            __br_handle_local_finish(skb);
            return RX_HANDLER_PASS;

        case 0x01:  /* IEEE MAC Pause */
            goto drop;

        case 0x0E:  /* LLDP */
            if (fwd_mask & (1u << dest[5]))
                goto forward;
            *pskb = skb;
            __br_handle_local_finish(skb);
            return RX_HANDLER_PASS;

        default:
            if (fwd_mask & (1u << dest[5]))
                goto forward;
        }

        /* 传递给本地协议栈 */
        if (NF_HOOK(NFPROTO_BRIDGE, NF_BR_LOCAL_IN,
                    dev_net(skb->dev), NULL, skb, skb->dev, NULL,
                    br_handle_local_finish) == 1) {
            return RX_HANDLER_PASS;
        } else {
            return RX_HANDLER_CONSUMED;
        }
    }

    /* 8. 处理特殊帧类型 */
    if (unlikely(br_process_frame_type(p, skb)))
        return RX_HANDLER_PASS;

forward:
    /* 9. 根据端口状态决定处理方式 */
    switch (p->state) {
    case BR_STATE_FORWARDING:
    case BR_STATE_LEARNING:
        /* 如果目的MAC是Bridge自己的MAC */
        if (ether_addr_equal(p->br->dev->dev_addr, dest))
            skb->pkt_type = PACKET_HOST;

        /* 调用Netfilter钩子，然后调用br_handle_frame_finish */
        return nf_hook_bridge_pre(skb, pskb);
        
    default:
drop:
        kfree_skb(skb);
    }
    
    return RX_HANDLER_CONSUMED;
}
```

### 5.2 br_handle_frame_finish()函数

```c
/*
 * br_handle_frame_finish - 完成数据包的接收处理
 * 
 * 执行MAC学习、转发决策等核心逻辑
 */
int br_handle_frame_finish(struct net *net, struct sock *sk, struct sk_buff *skb)
{
    struct net_bridge_port *p = br_port_get_rcu(skb->dev);
    enum br_pkt_type pkt_type = BR_PKT_UNICAST;
    struct net_bridge_fdb_entry *dst = NULL;
    bool local_rcv, mcast_hit = false;
    struct net_bridge *br;
    bool promisc;
    u16 vid = 0;
    u8 state;

    /* 1. 检查端口有效性 */
    if (!p || p->state == BR_STATE_DISABLED)
        goto drop;

    state = p->state;
    
    /* 2. VLAN过滤检查 */
    if (!br_allowed_ingress(p->br, nbp_vlan_group_rcu(p), skb, &vid,
                            &state, &vlan))
        goto out;

    /* 3. MAC学习（学习源MAC地址） */
    br = p->br;
    if (p->flags & BR_LEARNING)
        br_fdb_update(br, p, eth_hdr(skb)->h_source, vid, 0);

    promisc = !!(br->dev->flags & IFF_PROMISC);
    local_rcv = promisc;

    /* 4. 确定包类型 */
    if (is_multicast_ether_addr(eth_hdr(skb)->h_dest)) {
        if (is_broadcast_ether_addr(eth_hdr(skb)->h_dest)) {
            pkt_type = BR_PKT_BROADCAST;
            local_rcv = true;
        } else {
            pkt_type = BR_PKT_MULTICAST;
            /* 处理组播 */
            if (br_multicast_rcv(&brmctx, &pmctx, vlan, skb, vid))
                goto drop;
        }
    }

    /* 5. 如果是Learning状态，不能转发 */
    if (state == BR_STATE_LEARNING)
        goto drop;

    BR_INPUT_SKB_CB(skb)->brdev = br->dev;

    /* 6. 处理ARP/ND代理 */
    if (IS_ENABLED(CONFIG_INET) &&
        (skb->protocol == htons(ETH_P_ARP) ||
         skb->protocol == htons(ETH_P_RARP))) {
        br_do_proxy_suppress_arp(skb, br, vid, p);
    }

    /* 7. 根据包类型进行转发决策 */
    switch (pkt_type) {
    case BR_PKT_MULTICAST:
        /* 查找组播组 */
        mdst = br_mdb_get(brmctx, skb, vid);
        if ((mdst || BR_INPUT_SKB_CB_MROUTERS_ONLY(skb)) &&
            br_multicast_querier_exists(brmctx, eth_hdr(skb), mdst)) {
            if ((mdst && mdst->host_joined) ||
                br_multicast_is_router(brmctx, skb)) {
                local_rcv = true;
                DEV_STATS_INC(br->dev, multicast);
            }
            mcast_hit = true;
        } else {
            local_rcv = true;
            DEV_STATS_INC(br->dev, multicast);
        }
        break;
        
    case BR_PKT_UNICAST:
        /* 查找MAC地址表 */
        dst = br_fdb_find_rcu(br, eth_hdr(skb)->h_dest, vid);
        break;
        
    default:
        break;
    }

    /* 8. 执行转发 */
    if (dst) {
        unsigned long now = jiffies;

        /* 如果是本地地址，传递给本地协议栈 */
        if (test_bit(BR_FDB_LOCAL, &dst->flags))
            return br_pass_frame_up(skb, false);

        /* 更新使用时间 */
        if (now != dst->used)
            dst->used = now;
        
        /* 转发到目标端口 */
        br_forward(dst->dst, skb, local_rcv, false);
    } else {
        /* 未找到，泛洪 */
        if (!mcast_hit)
            br_flood(br, skb, pkt_type, local_rcv, false);
        else
            br_multicast_flood(mdst, skb, brmctx, local_rcv, false);
    }

    /* 9. 如果需要，传递给本地协议栈 */
    if (local_rcv)
        return br_pass_frame_up(skb, promisc);

out:
    return 0;
    
drop:
    kfree_skb(skb);
    goto out;
}
```

## 六、MAC学习机制

### 6.1 br_fdb_update()函数

```c
/* net/bridge/br_fdb.c */

/*
 * br_fdb_update - 更新MAC地址表
 * 
 * 学习源MAC地址，建立MAC→Port映射
 */
void br_fdb_update(struct net_bridge *br, struct net_bridge_port *source,
                   const unsigned char *addr, u16 vid, unsigned long flags)
{
    struct net_bridge_fdb_entry *fdb;

    /* 1. 查找MAC地址表 */
    fdb = fdb_find_rcu(&br->fdb_hash_tbl, addr, vid);
    
    if (likely(fdb)) {
        /* 2. 条目已存在，更新 */
        
        /* 如果是静态条目，不更新 */
        if (test_bit(BR_FDB_STATIC, &fdb->flags))
            return;
        
        /* 如果端口改变，更新端口 */
        if (fdb->dst != source) {
            fdb->dst = source;
            fdb->updated = jiffies;
        }
        
        /* 更新最后使用时间 */
        fdb->used = jiffies;
    } else {
        /* 3. 条目不存在，创建新条目 */
        spin_lock(&br->hash_lock);
        fdb = fdb_create(br, source, addr, vid, 0);
        spin_unlock(&br->hash_lock);
    }
}
```

```c
/*
 * MAC地址表老化机制

/* 老化时间计算 */
static inline unsigned long hold_time(const struct net_bridge *br)
{
    /* 如果拓扑变化中，使用forward_delay（默认15秒） */
    /* 否则使用ageing_time（默认300秒） */
    return br->topology_change ? br->forward_delay : br->ageing_time;
}

/* 检查条目是否过期 */
static inline int has_expired(const struct net_bridge *br,
                              const struct net_bridge_fdb_entry *fdb)
{
    return !test_bit(BR_FDB_STATIC, &fdb->flags) &&
           !test_bit(BR_FDB_ADDED_BY_EXT_LEARN, &fdb->flags) &&
           time_before_eq(fdb->updated + hold_time(br), jiffies);
}

/* 定期清理过期条目（由垃圾回收调用） */
void br_fdb_cleanup(struct work_struct *work)
{
    struct net_bridge *br = container_of(work, struct net_bridge, 
                                         gc_work.work);
    struct net_bridge_fdb_entry *f;
    unsigned long delay = hold_time(br);
    unsigned long now = jiffies;

    /* 遍历所有条目 */
    hlist_for_each_entry_safe(f, tmp, &br->fdb_list, fdb_node) {
        /* 跳过静态条目 */
        if (test_bit(BR_FDB_STATIC, &fdb->flags))
            continue;
        
        /* 检查是否过期 */
        if (time_before_eq(f->updated + delay, now))
            fdb_delete(br, f);
    }
    
    /* 重新调度垃圾回收 */
    mod_delayed_work(system_long_wq, &br->gc_work, delay);
}
```

```c
/*
 * br_become_root_bridge - 成为根桥
 */
void br_become_root_bridge(struct net_bridge *br)
{
    /* 1. 设置时间参数 */
    br->max_age = br->bridge_max_age;
    br->hello_time = br->bridge_hello_time;
    br->forward_delay = br->bridge_forward_delay;
    
    /* 2. 检测拓扑变化 */
    br_topology_change_detection(br);
    
    /* 3. 停止TCN定时器 */
    del_timer(&br->tcn_timer);
    
    /* 4. 如果Bridge已启动 */
    if (br->dev->flags & IFF_UP) {
        /* 发送配置BPDU */
        br_config_bpdu_generation(br);
        
        /* 启动Hello定时器 */
        mod_timer(&br->hello_timer, jiffies + br->hello_time);
    }
}

/*
 * br_port_state_selection - 选择端口状态
 */
void br_port_state_selection(struct net_bridge *br)
{
    struct net_bridge_port *p;

    /* 遍历所有端口 */
    list_for_each_entry(p, &br->port_list, list) {
        /* 如果是根端口 */
        if (p->port_no == br->root_port) {
            /* 设置为转发状态 */
            p->state = BR_STATE_FORWARDING;
            br_port_state_selection(p);
        } else if (br_is_designated_port(p)) {
            /* 如果是指定端口 */
            p->state = BR_STATE_FORWARDING;
        } else {
            /* 否则设置为阻塞状态 */
            p->state = BR_STATE_BLOCKING;
        }
    }
}
```

## 八、VLAN概述

### 8.1 什么是VLAN

**VLAN（Virtual Local Area Network，虚拟局域网）** 是一种将物理网络划分为多个逻辑网络的技术。

简单来说，VLAN就是**网络隔离技术**：
- 在同一物理网络上创建多个虚拟网络
- 不同VLAN之间二层隔离
- 通过VLAN标签（802.1Q）区分

### 9.1 vlan_dev_priv结构

```c
 * 每个VLAN子接口（如eth0.100）对应一个vlan_dev_priv
```

## 十、VLAN子接口实现

### 10.1 VLAN子接口创建

```c
/* net/8021q/vlan.c */

/*
 * register_vlan_dev - 注册VLAN设备
 */
int register_vlan_dev(struct net_device *dev, struct netlink_ext_ack *extack)
{
    struct vlan_dev_priv *vlan = vlan_dev_priv(dev);
    struct net_device *real_dev = vlan->real_dev;
    u16 vlan_id = vlan->vlan_id;
    struct vlan_info *vlan_info;
    struct vlan_group *grp;
    int err;

    /* 1. 添加VLAN ID到真实设备 */
    err = vlan_vid_add(real_dev, vlan->vlan_proto, vlan_id);
    if (err)
        return err;

    vlan_info = rtnl_dereference(real_dev->vlan_info);
    grp = &vlan_info->grp;
    
    /* 2. 初始化GVRP/MVRP */
    if (grp->nr_vlan_devs == 0) {
        err = vlan_gvrp_init_applicant(real_dev);
        if (err < 0)
            goto out_vid_del;
        err = vlan_mvrp_init_applicant(real_dev);
        if (err < 0)
            goto out_uninit_gvrp;
    }

    /* 3. 预分配VLAN数组 */
    err = vlan_group_prealloc_vid(grp, vlan->vlan_proto, vlan_id);
    if (err < 0)
        goto out_uninit_mvrp;

    /* 4. 注册网络设备 */
    err = register_netdevice(dev);
    if (err < 0)
        goto out_uninit_mvrp;

    /* 5. 建立upper/lower链接 */
    err = netdev_upper_dev_link(real_dev, dev, extack);
    if (err)
        goto out_unregister_netdev;

    /* 6. 更新操作状态 */
    vlan_stacked_transfer_operstate(real_dev, dev, vlan);
    linkwatch_fire_event(dev);

    /* 7. 将VLAN设备加入组 */
    vlan_group_set_device(grp, vlan->vlan_proto, vlan_id, dev);
    grp->nr_vlan_devs++;

    netdev_update_features(dev);

    return 0;
}
```

### 10.2 VLAN子接口发送

```c
/* net/8021q/vlan_dev.c */

/*
 * vlan_dev_hard_start_xmit - VLAN设备发送函数
 */
static netdev_tx_t vlan_dev_hard_start_xmit(struct sk_buff *skb,
                                            struct net_device *dev)
{
    struct vlan_dev_priv *vlan = vlan_dev_priv(dev);
    struct vlan_ethhdr *veth = (struct vlan_ethhdr *)(skb->data);
    unsigned int len;
    int ret;

    /* 1. 如果需要，添加VLAN标签 */
    if (vlan->flags & VLAN_FLAG_REORDER_HDR ||
        veth->h_vlan_proto != vlan->vlan_proto) {
        u16 vlan_tci;
        
        /* 构造VLAN TCI */
        vlan_tci = vlan->vlan_id;
        vlan_tci |= vlan_dev_get_egress_qos_mask(dev, skb->priority);
        
        /* 添加VLAN标签到skb */
        __vlan_hwaccel_put_tag(skb, vlan->vlan_proto, vlan_tci);
    }

    /* 2. 切换到真实设备 */
    skb->dev = vlan->real_dev;
    len = skb->len;
    
    /* 3. 发送 */
    ret = dev_queue_xmit(skb);

    /* 4. 更新统计 */
    if (likely(ret == NET_XMIT_SUCCESS || ret == NET_XMIT_CN)) {
        struct vlan_pcpu_stats *stats;

        stats = this_cpu_ptr(vlan->vlan_pcpu_stats);
        u64_stats_update_begin(&stats->syncp);
        stats->tx_packets++;
        stats->tx_bytes += len;
        u64_stats_update_end(&stats->syncp);
    } else {
        this_cpu_inc(vlan->vlan_pcpu_stats->tx_dropped);
    }

    return ret;
}
```

## 十一、Bridge VLAN过滤

### 11.1 Bridge VLAN过滤使能

```c
/* net/bridge/br_vlan.c */

/*
 * br_vlan_enabled - 检查Bridge VLAN过滤是否使能
 */
bool br_vlan_enabled(const struct net_device *dev)
{
    struct net_bridge *br;

    if (!netif_is_bridge_master(dev))
        return false;

    br = netdev_priv(dev);
    return br_opt_get(br, BROPT_VLAN_ENABLED);
}

/*
 * __br_vlan_set_proto - 设置VLAN协议
 */
int __br_vlan_set_proto(struct net_bridge *br, __be16 proto)
{
    /* 检查协议有效性 */
    if (proto != htons(ETH_P_8021Q) && proto != htons(ETH_P_8021AD))
        return -EINVAL;

    /* 更新协议 */
    br->vlan_proto = proto;
    
    return 0;
}
```

### 11.2 Bridge VLAN入口过滤

```c
/*
 * br_allowed_ingress - 检查入口VLAN是否允许
 */
bool br_allowed_ingress(const struct net_bridge *br,
                        struct net_bridge_vlan_group *vg,
                        struct sk_buff *skb, u16 *pvid,
                        u8 *state, struct net_bridge_vlan **vlan)
{
    u16 vid;
    
    /* 1. 如果VLAN过滤未使能，允许所有 */
    if (!br_opt_get(br, BROPT_VLAN_ENABLED))
        return true;

    /* 2. 获取PVID */
    *pvid = vg->pvid;
    *state = vg->pvid_state;

    /* 3. 提取VLAN ID */
    if (!skb_vlan_tag_present(skb)) {
        /* 无VLAN标签，使用PVID */
        vid = *pvid;
        if (!vid)
            return false;  /* 丢弃 */
    } else {
        /* 有VLAN标签 */
        vid = skb_vlan_tag_get_id(skb);
    }

    /* 4. 查找VLAN条目 */
    *vlan = br_vlan_find(vg, vid);
    if (!*vlan)
        return false;  /* VLAN不存在，丢弃 */

    /* 5. 检查VLAN状态 */
    *state = (*vlan)->state;
    if (*state == BR_STATE_BLOCKING)
        return false;

    return true;
}
```

### 11.3 Bridge VLAN出口过滤

```c
/*
 * br_allowed_egress - 检查出口VLAN是否允许
 */
bool br_allowed_egress(struct net_bridge_vlan_group *vg,
                       const struct sk_buff *skb)
{
    const struct net_bridge_vlan *vlan;
    u16 vid;

    /* 1. 获取VLAN ID */
    if (!skb_vlan_tag_present(skb))
        vid = vg->pvid;
    else
        vid = skb_vlan_tag_get_id(skb);

    /* 2. 查找VLAN条目 */
    vlan = br_vlan_find(vg, vid);
    if (!vlan)
        return false;

    /* 3. 检查是否为untagged */
    if (vlan->flags & BRIDGE_VLAN_INFO_UNTAGGED) {
        /* 剥掉VLAN标签 */
        skb->vlan_present = 0;
    }

    return true;
}
```

### 12.4 Docker网络配置详解

```bash
# 1. Docker创建bridge网络
#    创建docker0 bridge设备
ip link add name docker0 type bridge
ip addr add 172.17.0.1/16 dev docker0
ip link set docker0 up

# 2. 启动容器时
#    创建veth pair
ip link add veth123 type veth peer name eth0

#    将一端移到容器命名空间
ip link set eth0 netns <容器PID>

#    在容器内配置IP
ip netns exec <容器PID> ip addr add 172.17.0.2/16 dev eth0
ip netns exec <容器PID> ip link set eth0 up

#    将另一端连接到docker0
ip link set veth123 master docker0
ip link set veth123 up

# 3. 配置NAT规则
#    允许容器访问外部网络
iptables -t nat -A POSTROUTING -s 172.17.0.0/16 ! -o docker0 -j MASQUERADE

#    允许外部访问容器端口（端口映射）
iptables -t nat -A PREROUTING -m addrtype --dst-type LOCAL -j DOCKER
iptables -t nat -A DOCKER ! -i docker0 -p tcp --dport 8080 -j DNAT --to-destination 172.17.0.2:80

# 4. 查看网络配置
#    查看bridge
ip link show docker0
bridge link show

#    查看容器网络
docker inspect <容器ID> | grep -A 20 "NetworkSettings"

#    查看NAT规则
iptables -t nat -L -n -v
```

### 掌握标准检验

**能用br_handle_frame()转发逻辑解释Docker默认的bridge模式网络是如何让容器上网的：**

✅ **已掌握：**

1. **Bridge是软件交换机**：docker0是一个Bridge设备，实现二层转发功能

2. **veth pair连接容器**：每个容器通过veth pair连接到docker0，一端在容器内（eth0），一端在主机上（vethxxx）

3. **br_handle_frame()转发流程**：
   - 容器发送包 → veth pair → docker0接收
   - br_handle_frame()处理 → MAC学习
   - br_handle_frame_finish() → 转发决策
   - 如果目的MAC在本地 → br_forward()转发到对应容器
   - 如果目的MAC不在本地 → 传递给主机协议栈

4. **容器上网流程**：
   - 容器发送包（源IP：容器IP，目的IP：外部IP）
   - docker0接收 → 查找路由表 → 目的不在本地网络
   - 转发给默认网关（docker0自己）
   - 主机协议栈处理 → iptables NAT转换
   - 源IP替换为主机IP → 通过物理网卡发送
   - 响应包返回 → NAT反向转换 → 通过docker0返回容器

5. **关键机制**：
   - MAC学习：docker0自动学习容器MAC地址
   - 二层转发：容器间通信无需经过主机协议栈
   - NAT：容器访问外部网络通过iptables NAT
   - 端口映射：外部访问容器通过iptables DNAT

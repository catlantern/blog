## 二、Jiffies：系统时间的心跳

### 2.1 什么是Jiffies

**Jiffies是Linux内核中记录系统启动以来经过的tick数的全局变量**，它是整个内核时间系统的基石。

定义在 **`include/linux/jiffies.h`**：

```c
/*
 * jiffies - 32位tick计数器
 * 每次tick中断时递增
 */
extern unsigned long volatile __cacheline_aligned_in_smp __jiffy_arch_data jiffies;

/*
 * jiffies_64 - 64位tick计数器
 * 避免长时间运行后溢出
 */
extern u64 __cacheline_aligned_in_smp jiffies_64;
```

### 2.2 HZ：节拍率

**HZ定义了每秒的tick数**，决定了tick中断的频率：

```
┌─────────────────────────────────────────────────────────────┐
│                    HZ配置与精度                              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  常见HZ值：                                                  │
│  ┌─────────────────────────────────────────────────────┐   │
│  │ HZ值    │ tick间隔    │ 精度     │ 应用场景          │   │
│  ├─────────────────────────────────────────────────────┤   │
│  │ 100     │ 10ms        │ 低       │ 服务器            │   │
│  │ 250     │ 4ms         │ 中       │ 桌面系统          │   │
│  │ 1000    │ 1ms         │ 高       │ 实时系统          │   │
│  │ 1024    │ ~0.98ms     │ 高       │ 某些嵌入式        │   │
│  └─────────────────────────────────────────────────────┘   │
│                                                             │
│  配置位置：                                                  │
│    - 内核配置：CONFIG_HZ                                     │
│    - 架构相关：arch/<arch>/include/asm/param.h              │
│                                                             │
│  时间转换：                                                  │
│    - 1秒 = HZ个jiffies                                      │
│    - 1毫秒 = HZ/1000个jiffies                               │
│    - jiffies_to_msecs(j) = j * 1000 / HZ                    │
│    - msecs_to_jiffies(m) = m * HZ / 1000                    │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 2.3 Jiffies的关键宏

```c
/*
 * 时间比较宏 - 正确处理jiffies溢出
 */

/* 判断时间a是否在时间b之后 */
#define time_after(a, b)    ((long)((b) - (a)) < 0)

/* 判断时间a是否在时间b之前 */
#define time_before(a, b)   time_after(b, a)

/* 判断时间a是否在时间b之后或等于 */
#define time_after_eq(a, b) ((long)((a) - (b)) >= 0)

/* 判断时间a是否在时间b之前或等于 */
#define time_before_eq(a, b) time_after_eq(b, a)

/*
 * 使用示例：
 * 错误方式：if (jiffies > timeout)  // 溢出后会出错
 * 正确方式：if (time_after(jiffies, timeout))  // 正确处理溢出
 */
```

### 2.4 时间转换函数

```c
/*
 * jiffies与毫秒/微秒转换
 */

/* jiffies转毫秒 */
unsigned int jiffies_to_msecs(const unsigned long j);

/* jiffies转微秒 */
unsigned int jiffies_to_usecs(const unsigned long j);

/* 毫秒转jiffies */
unsigned long msecs_to_jiffies(const unsigned int m);

/* 微秒转jiffies */
unsigned long usecs_to_jiffies(const unsigned int u);

/*
 * 使用示例：
 */
void example_time_conversion(void)
{
    unsigned long timeout;
    
    /* 设置100毫秒后的超时 */
    timeout = jiffies + msecs_to_jiffies(100);
    
    /* 检查是否超时 */
    if (time_after(jiffies, timeout)) {
        pr_info("Timeout!\n");
    }
    
    /* 将jiffies转换为毫秒显示 */
    unsigned long uptime_ms = jiffies_to_msecs(jiffies);
    pr_info("System uptime: %lu ms\n", uptime_ms);
}
```

## 三、Tick定时器：内核的节拍器

### 3.1 什么是Tick定时器

**Tick定时器是内核周期性产生中断的核心机制**，它驱动着整个系统的运转：

```
┌─────────────────────────────────────────────────────────────┐
│                    Tick定时器的作用                          │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  每个tick中断发生时，内核执行：                               │
│                                                             │
│  1. 更新jiffies                                             │
│     jiffies_64++                                            │
│                                                             │
│  2. 更新系统时间                                             │
│     update_wall_time()                                      │
│                                                             │
│  3. 触发进程调度                                             │
│     scheduler_tick()                                        │
│     - 检查时间片是否用完                                     │
│     - 更新运行队列                                           │
│     - 可能触发进程切换                                       │
│                                                             │
│  4. 处理定时器                                               │
│     run_local_timers()                                      │
│     - 检查并执行到期的定时器                                 │
│                                                             │
│  5. 更新统计信息                                             │
│     - CPU使用率                                             │
│     - 负载平均值                                            │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 3.2 Tick的工作模式

```
│  ┌─────────────────────────────────────────────────────┐   │

│  │                                                     │   │

│  ┌─────────────────────────────────────────────────────┐   │

│  │       支持纳秒级精度                                 │   │
│  │ 优点：精度高，适合实时应用                           │   │
│  │ 缺点：开销较大                                       │   │
│  └─────────────────────────────────────────────────────┘   │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 3.4 scheduler_tick函数

定义在 **`kernel/sched/core.c`**：

```c
/*
 * scheduler_tick - 每个tick调用一次
 * 
 * 负责更新调度相关信息，可能触发进程切换
 */
void scheduler_tick(void)
{
    int cpu = smp_processor_id();
    struct rq *rq = cpu_rq(cpu);
    struct task_struct *curr = rq->curr;
    struct rq_flags rf;

    arch_scale_freq_tick();      /* 频率缩放 */
    sched_clock_tick();          /* 更新调度时钟 */

    rq_lock(rq, &rf);

    update_rq_clock(rq);         /* 更新运行队列时钟 */
    
    /* 调用调度类的task_tick方法 */
    curr->sched_class->task_tick(rq, curr, 0);
    
    calc_global_load_tick(rq);   /* 更新全局负载 */

    rq_unlock(rq, &rf);

    perf_event_task_tick();      /* 性能事件 */
}
```

### 4.2 用户态定时器示例

```c
/*
 * 示例1：使用sleep简单延时
 */
void example_sleep(void)
{
    printf("Start\n");
    sleep(2);  /* 睡眠2秒 */
    printf("After 2 seconds\n");
}

/*
 * 示例2：使用nanosleep高精度延时
 */
void example_nanosleep(void)
{
    struct timespec ts;
    ts.tv_sec = 0;
    ts.tv_nsec = 500000000;  /* 500毫秒 */
    
    nanosleep(&ts, NULL);
}

/*
 * 示例3：使用POSIX定时器
 */
#include <signal.h>
#include <time.h>

timer_t timerid;

void timer_handler(int sig, siginfo_t *si, void *uc)
{
    printf("Timer expired!\n");
}

void example_posix_timer(void)
{
    struct sigevent sev;
    struct itimerspec its;
    struct sigaction sa;

    /* 设置信号处理函数 */
    sa.sa_flags = SA_SIGINFO;
    sa.sa_sigaction = timer_handler;
    sigaction(SIGRTMIN, &sa, NULL);

    /* 创建定时器 */
    sev.sigev_notify = SIGEV_SIGNAL;
    sev.sigev_signo = SIGRTMIN;
    sev.sigev_value.sival_ptr = &timerid;
    timer_create(CLOCK_MONOTONIC, &sev, &timerid);

    /* 设置定时器：1秒后首次触发，之后每2秒触发一次 */
    its.it_value.tv_sec = 1;
    its.it_value.tv_nsec = 0;
    its.it_interval.tv_sec = 2;
    its.it_interval.tv_nsec = 0;
    timer_settime(timerid, 0, &its, NULL);
}

/*
 * 示例4：使用timerfd
 */
#include <sys/timerfd.h>
#include <sys/epoll.h>

void example_timerfd(void)
{
    int tfd, epfd;
    struct itimerspec its;
    struct epoll_event ev;

    /* 创建timerfd */
    tfd = timerfd_create(CLOCK_MONOTONIC, 0);
    
    /* 设置定时器 */
    its.it_value.tv_sec = 1;
    its.it_value.tv_nsec = 0;
    its.it_interval.tv_sec = 1;
    its.it_interval.tv_nsec = 0;
    timerfd_settime(tfd, 0, &its, NULL);

    /* 使用epoll监听 */
    epfd = epoll_create1(0);
    ev.events = EPOLLIN;
    ev.data.fd = tfd;
    epoll_ctl(epfd, EPOLL_CTL_ADD, tfd, &ev);

    /* 等待定时器 */
    while (1) {
        epoll_wait(epfd, &ev, 1, -1);
        uint64_t expirations;
        read(tfd, &expirations, sizeof(expirations));
        printf("Timer expired %lu times\n", expirations);
    }
}
```

### 5.2 精度对比

| 定时类型 | 精度 | 来源 | 适用场景 |
|----------|------|------|----------|
| Tick | 1/HZ秒（如1ms） | 硬件中断 | 调度、统计 |
| Jiffies | 1/HZ秒 | Tick累加 | 时间基准 |
| 内核定时器 | 1/HZ秒 | Jiffies | 驱动延时 |

| 用户态定时器 | 取决于时钟源 | 系统调用 | 应用程序 |

### 7.2 tick_periodic函数

定义在 **`kernel/time/tick-common.c`**：

```c
/*
 * tick_periodic - 周期性tick处理
 * 
 * 每个tick中断时调用
 */
static void tick_periodic(int cpu)
{
    /* 只有一个CPU负责更新jiffies */
    if (tick_do_timer_cpu == cpu) {
        raw_spin_lock(&jiffies_lock);
        write_seqcount_begin(&jiffies_seq);

        /* 更新下一个tick时间 */
        tick_next_period = ktime_add_ns(tick_next_period, TICK_NSEC);

        /* 更新jiffies */
        do_timer(1);
        
        write_seqcount_end(&jiffies_seq);
        raw_spin_unlock(&jiffies_lock);
        
        /* 更新系统时间 */
        update_wall_time();
    }

    /* 更新进程时间统计 */
    update_process_times(user_mode(get_irq_regs()));
    
    /* 性能分析采样 */
    profile_tick(CPU_PROFILING);
}
```

```c
 * update_process_times - 更新进程时间

{

    /* 统计CPU时间 */
    account_process_tick(p, user_tick);
    
    /* 运行本地定时器 */
```

### 8.1 定时器初始化与使用

```c
#include <linux/timer.h>
#include <linux/jiffies.h>

/* 定义定时器 */
struct timer_list my_timer;

/* 定时器回调函数 */
void my_timer_callback(struct timer_list *t)
{
    struct my_data *data = from_timer(data, t, my_timer);
    
    pr_info("Timer expired at jiffies=%lu\n", jiffies);

void setup_my_timer(void)
{
    /* 方法1：使用timer_setup（推荐） */
    timer_setup(&my_timer, my_timer_callback, 0);
    
    /* 设置到期时间 */
    my_timer.expires = jiffies + msecs_to_jiffies(1000);
    
    /* 添加定时器 */
    add_timer(&my_timer);
}

/* 修改定时器 */
void modify_my_timer(void)
{
    /* 修改到期时间为500ms后 */
    mod_timer(&my_timer, jiffies + msecs_to_jiffies(500));
}

/* 删除定时器 */
void cleanup_my_timer(void)
{
    /* 同步删除（等待回调完成） */
    del_timer_sync(&my_timer);
}
```

### 8.3 完整示例：内核模块定时器

```c
#include <linux/module.h>
#include <linux/timer.h>
#include <linux/jiffies.h>

#define TIMER_INTERVAL_MS  1000

static struct timer_list my_timer;
static int timer_count = 0;

/*
 * 定时器回调函数
 * 注意：运行在中断上下文，不能睡眠
 */
static void my_timer_callback(struct timer_list *t)
{
    timer_count++;
    
    pr_info("Timer callback: count=%d, jiffies=%lu\n", 
            timer_count, jiffies);
    
    /* 重新设置定时器，实现周期性执行 */
    mod_timer(&my_timer, jiffies + msecs_to_jiffies(TIMER_INTERVAL_MS));
}

static int __init my_module_init(void)
{
    pr_info("Timer module loaded\n");
    
    /* 初始化并启动定时器 */
    timer_setup(&my_timer, my_timer_callback, 0);
    mod_timer(&my_timer, jiffies + msecs_to_jiffies(TIMER_INTERVAL_MS));
    
    return 0;
}

static void __exit my_module_exit(void)
{
    /* 同步删除定时器 */
    del_timer_sync(&my_timer);
    
    pr_info("Timer module unloaded, total callbacks=%d\n", timer_count);
}

module_init(my_module_init);
module_exit(my_module_exit);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Example");
MODULE_DESCRIPTION("Kernel timer example");
```

## 总结

Tick定时器、Jiffies和用户态定时器构成了Linux时间系统的核心：

1. **Jiffies**：系统启动以来的tick计数，时间系统的基石
2. **Tick定时器**：周期性产生中断，驱动整个系统运转
3. **用户态定时器**：通过系统调用使用内核定时功能

**关键关系**：
- Tick中断 → Jiffies递增 → 定时器检查 → 进程唤醒
- HZ决定tick频率和精度
- time_after宏正确处理jiffies溢出

**关键要点**：
- Tick是内核的"心跳"，驱动调度和定时
- Jiffies是时间基准，所有定时器基于它
- 用户态定时器通过系统调用使用内核定时功能
- 定时器回调运行在中断上下文，不能睡眠
- NOHZ模式可减少空闲时的tick中断

## 一、从一个问题开始：数据如何传输？

## 二、协议注册机制

### 2.1 协议如何被识别？

当 IP 层收到数据包时，需要知道该交给哪个传输层协议处理。

```c
/*
 * net_protocol - 网络协议注册结构
 * 定义在 include/net/protocol.h
 */
struct net_protocol {
    int (*handler)(struct sk_buff *skb);      /* 协议处理函数 */
    int (*err_handler)(struct sk_buff *skb, u32 info); /* 错误处理函数 */
    unsigned int no_policy:1;                  /* 不需要策略检查 */
};

/* 协议注册表 */
struct net_protocol __rcu *inet_protos[MAX_INET_PROTOS] __read_mostly;
```

```
协议注册表结构：
+===========================================================================+
|                                                                           |
|   inet_protos[] 数组：                                                    |
|   +-------------------+-------------------+                                |
|   | 索引 (协议号)    | 注册的结构        |                                |
|   +-------------------+-------------------+                                |
|   | IPPROTO_TCP (6)  | tcp_protocol      |  TCP 协议                     |
|   | IPPROTO_UDP (17) | udp_protocol      |  UDP 协议                     |
|   | IPPROTO_ICMP (1) | icmp_protocol     |  ICMP 协议                    |
|   | IPPROTO_RAW (255)| raw_protocol      |  原始套接字                   |

```

```c
 * TCP 协议注册
```

### 2.4 协议注册函数

```c
/*
 * inet_add_protocol - 注册协议
 * 定义在 net/ipv4/protocol.c
 */
int inet_add_protocol(const struct net_protocol *prot, unsigned char protocol)
{
    /* 使用原子操作注册协议 */
    return !cmpxchg((const struct net_protocol **)&inet_protos[protocol],
                    NULL, prot) ? 0 : -1;
}
```

## 三、TCP 协议实现

### 3.1 TCP 协议的特点

```
TCP 协议特点：
+===========================================================================+
|                                                                           |
|   1. 面向连接                                                             |
|   +-------------------+                                                    |
|   | 三次握手建立连接 |  SYN → SYN+ACK → ACK                             |
|   | 四次挥手断开连接 |  FIN → ACK → FIN → ACK                           |

|                                                                           |

|   +-------------------+                                                    |
|   | 确认应答         |  每个数据包都有 ACK                              |

|                                                                           |

|   +-------------------+                                                    |

|   | 拥塞避免         |  线性增长拥塞窗口                                |
|   | 快速重传         |  收到 3 个重复 ACK 立即重传                      |
|   | 快速恢复         |  快速重传后的恢复算法                            |
|   +-------------------+                                                    |
|                                                                           |
+===========================================================================+
```

### 3.4 TCP 连接建立：三次握手

```
TCP 三次握手过程：
+===========================================================================+
|                                                                           |
|   客户端                              服务器                              |
|   (CLOSED)                            (LISTEN)                            |
|       |                                   |                               |
|       | ---- SYN (seq=x) ---------------> |                               |
|       |     状态: SYN_SENT                |                               |
|       |                                   |                               |
|       | <--- SYN+ACK (seq=y, ack=x+1) --- |                               |
|       |     状态: ESTABLISHED             | 状态: SYN_RECV                |
|       |                                   |                               |
|       | ---- ACK (ack=y+1) -------------> |                               |
|       |                                   | 状态: ESTABLISHED             |
|       |                                   |                               |
|   数据传输开始                         数据传输开始                       |
|                                                                           |
+===========================================================================+
```

```c
/*
 * TCP 连接请求处理
 * 定义在 net/ipv4/tcp_ipv4.c
 */
int tcp_v4_conn_request(struct sock *sk, struct sk_buff *skb)
{
    /* 1. 检查是否需要发送 SYN Cookie */
    if (inet_csk_reqsk_queue_is_full(sk) && !isn) {
        want_cookie = tcp_syn_flood_action(sk, rsk_ops->slab_name);
        if (!want_cookie)
            goto drop;
    }
    
    /* 2. 分配请求套接字 */
    req = inet_reqsk_alloc(rsk_ops, sk, want_cookie);
    if (!req)
        goto drop;
    
    /* 3. 初始化请求套接字 */
    tcp_rsk(req)->snt_isn = isn;
    tcp_rsk(req)->txhash = net_tx_rndhash();
    
    /* 4. 发送 SYN+ACK */
    if (tcp_v4_send_synack(sk, dst, req, skb, want_cookie))
        goto drop;
    
    return 0;
}
```

```c
/*
 * tcp_sendmsg - TCP 发送数据主函数
 * 定义在 net/ipv4/tcp.c
 */
int tcp_sendmsg(struct sock *sk, struct msghdr *msg, size_t size)
{
    struct tcp_sock *tp = tcp_sk(sk);
    int flags, err, copied = 0;
    long timeo;
    
    /* 1. 等待连接建立 */
    if (sk->sk_state != TCP_ESTABLISHED)
        timeo = sock_wait_for_wmt(sk, flags);
    
    /* 2. 获取发送参数 */
    mss_now = tcp_send_mss(sk, &size_goal, flags);
    
    /* 3. 循环发送数据 */
    while (msg_data_left(msg)) {
        /* 分配 skb 并填充数据 */
        skb = tcp_build_frag(sk, size_goal, flags, page, offset, &copy);
        
        /* 推送数据 */
        if (forced_push(tp) || skb == tcp_send_head(sk))
            tcp_push(sk, flags, mss_now, size_goal);
    }
    
    return copied;
}
```

```c
/*
 * tcp_v4_rcv - TCP 数据包接收入口
 * 定义在 net/ipv4/tcp_ipv4.c
 */
int tcp_v4_rcv(struct sk_buff *skb)
{
    struct tcphdr *th;
    struct sock *sk;
    
    /* 1. 检查数据包有效性 */
    if (!pskb_may_pull(skb, sizeof(struct tcphdr)))
        goto discard;
    
    th = tcp_hdr(skb);
    
    /* 2. 查找对应的套接字 */
    sk = __inet_lookup_skb(&tcp_hashinfo, skb, th->source, th->dest);
    if (!sk)
        goto no_tcp_socket;
    
    /* 3. 处理数据包 */
    if (sk->sk_state == TCP_TIME_WAIT)
        goto do_time_wait;
    
    if (sk->sk_state == TCP_NEW_SYN_RECV)
        goto handle_syn_recv;
    
    /* 4. 调用状态机处理 */
    tcp_rcv_established(sk, skb);
    
    return 0;
}
```

```
+===========================================================================+
|                                                                           |
```

```
|   +-------------------+                                                    |
|   | ssthresh = cwnd/2 |                                                    |

|   | 进入拥塞避免       |                                                  |
```

```c
/*
 * udp_sendmsg - UDP 发送数据主函数
 * 定义在 net/ipv4/udp.c
 */
int udp_sendmsg(struct sock *sk, struct msghdr *msg, size_t len)
{
    struct inet_sock *inet = inet_sk(sk);
    struct flowi4 *fl4;
    struct rtable *rt;
    int err;
    
    /* 1. 检查数据长度 */
    if (len > 0xFFFF)
        return -EMSGSIZE;
    
    /* 2. 确定目的地址和端口 */
    if (msg->msg_name) {
        /* 使用指定的地址 */
        daddr = usin->sin_addr.s_addr;
        dport = usin->sin_port;
    } else {
        /* 使用已连接的地址 */
        daddr = inet->inet_daddr;
        dport = inet->inet_dport;
    }
    
    /* 3. 路由查找 */
    rt = ip_route_output_ports(net, fl4, sk, daddr, saddr,
                               dport, inet->inet_sport, ...);
    
    /* 4. 发送数据 */
    if (corkreq) {
        /* 延迟发送模式 */
        err = udp_push_pending_frames(sk);
    } else {
        /* 立即发送 */
        err = udp_send_skb(skb, fl4, &inet->cork.base);
    }
    
    return err;
}
```

```c
int udp_rcv(struct sk_buff *skb)
{
    return __udp4_lib_rcv(skb, &udp_table, IPPROTO_UDP);
}

/*
 * __udp4_lib_rcv - UDP 接收处理
 */
int __udp4_lib_rcv(struct sk_buff *skb, struct udp_table *udptable,
                   int proto)
{
    struct udphdr *uh;
    struct sock *sk;
    
    /* 1. 验证 UDP 头部 */
    if (!pskb_may_pull(skb, sizeof(struct udphdr)))
        goto drop;
    
    uh = udp_hdr(skb);
    
    /* 2. 校验和验证 */
    if (udp4_csum_init(skb, uh, proto))
        goto csum_error;
    
    /* 3. 查找套接字 */
    sk = __udp4_lib_lookup_skb(skb, uh->source, uh->dest, udptable);
    
    /* 4. 放入接收队列 */
    if (sk) {
        udp_queue_rcv_skb(sk, skb);
        return 0;
    }
    
    /* 5. 没有找到套接字，发送 ICMP 端口不可达 */
    icmp_send(skb, ICMP_DEST_UNREACH, ICMP_PORT_UNREACH, 0);
    
drop:
    kfree_skb(skb);
    return 0;
}
```

```c
/*
 * ip_rcv - IP 数据包接收入口
 * 定义在 net/ipv4/ip_input.c

int ip_rcv(struct sk_buff *skb, struct net_device *dev,
           struct packet_type *pt, struct net_device *orig_dev)
{
    struct iphdr *iph;
    
    /* 1. 验证 IP 头部 */
    if (!pskb_may_pull(skb, sizeof(struct iphdr)))
        goto drop;
    
    iph = ip_hdr(skb);

    if (iph->ihl < 5 || iph->version != 4)
        goto inhdr_error;

    if (unlikely(ip_fast_csum((u8 *)iph, iph->ihl)))
        goto csum_error;

    len = ntohs(iph->tot_len);
    if (skb->len < len || len < (iph->ihl * 4))
        goto inhdr_error;
    
    /* 5. 调用 netfilter 和本地处理 */
    return NF_HOOK(NFPROTO_IPV4, NF_INET_PRE_ROUTING,
                   net, NULL, skb, dev, NULL,
                   ip_rcv_finish);
}
```

```c
/*
 * ip_do_fragment - IP 分片
 * 定义在 net/ipv4/ip_output.c
 */
int ip_do_fragment(struct net *net, struct sock *sk, struct sk_buff *skb,
                   int (*output)(struct net *, struct sock *, struct sk_buff *))
{
    struct iphdr *iph;
    unsigned int mtu, hlen, left, len;
    int offset;
    
    iph = ip_hdr(skb);
    mtu = ip_skb_dst_mtu(sk, skb);
    hlen = iph->ihl * 4;
    
    /* 分片 */
    left = skb->len - hlen;
    offset = (ntohs(iph->frag_off) & IP_OFFSET) << 3;
    
    while (left > 0) {
        len = left;
        if (len > mtu)
            len = mtu;
        
        /* 创建分片 */
        skb2 = alloc_skb(len + hlen + LL_RESERVED_SPACE(dev), GFP_ATOMIC);
        
        /* 复制 IP 头 */
        skb_copy_from_linear_data(skb, skb_network_header(skb2), hlen);
        
        /* 设置分片标志 */
        if (left > len)
            iph2->frag_off |= htons(IP_MF);
        
        /* 发送分片 */
        output(net, sk, skb2);
        
        left -= len;
        offset += len;
    }
    
    return 0;
}
```

```c
/*
 * icmp_rcv - ICMP 数据包接收
 * 定义在 net/ipv4/icmp.c
 */
int icmp_rcv(struct sk_buff *skb)
{
    struct icmphdr *icmph;
    
    /* 验证 ICMP 头部 */
    if (!pskb_may_pull(skb, sizeof(struct icmphdr)))
        goto drop;
    
    icmph = icmp_hdr(skb);
    
    /* 校验和验证 */
    if (!skb_csum_unnecessary(skb) && __skb_checksum_validate(skb, ...))
        goto csum_error;
    
    /* 根据类型分发处理 */
    switch (icmph->type) {
    case ICMP_ECHO:
        icmp_echo(skb);
        break;
    case ICMP_DEST_UNREACH:
    case ICMP_TIME_EXCEEDED:
    case ICMP_PARAMETERPROB:
        icmp_unreach(skb);
        break;
    default:
        break;
    }
    
    return 0;
}
```

## 九、总结

### 9.1 协议栈模块核心概念

```
协议栈模块核心概念：
+===========================================================================+
|                                                                           |
|   1. 协议分层                                                             |
|   +-------------------+                                                    |
|   | 传输层 (TCP/UDP) |  端到端通信                                      |
|   | 网络层 (IP)      |  主机到主机                                      |
|   +-------------------+                                                    |
|                                                                           |
|   2. 协议注册                                                             |
|   +-------------------+                                                    |
|   | net_protocol     |  协议处理函数注册                                |
|   | inet_protos[]    |  协议分发表                                      |
|   +-------------------+                                                    |
|                                                                           |
|   3. TCP 特性                                                             |
|   +-------------------+                                                    |
|   | 连接管理         |  三次握手、四次挥手                              |
|   | 可靠传输         |  确认、重传、序号                                |
|   | 流量控制         |  滑动窗口                                        |
|   | 拥塞控制         |  慢启动、拥塞避免、快速重传                      |
|   +-------------------+                                                    |
|                                                                           |
|   4. UDP 特性                                                             |
|   +-------------------+                                                    |
|   | 无连接           |  直接发送                                        |
|   | 不可靠           |  不保证送达                                      |
|   | 快速             |  低开销                                          |
|   +-------------------+                                                    |
|                                                                           |
|   5. IP 功能                                                              |
|   +-------------------+                                                    |
|   | 寻址             |  IP 地址                                         |
|   | 路由             |  路由表查找                                      |
|   | 分片             |  MTU 适配                                        |
|   +-------------------+                                                    |
|                                                                           |
+===========================================================================+
```
